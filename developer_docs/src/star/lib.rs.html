16pub fn model(
17    num_points: u64,
18    r1: f64,
19    r2: f64,
20    h: f64,
21    core: &mut fj::core::Core,
22) -> Solid {
23    let num_vertices = num_points * 2;
24    let vertex_iter = (0..num_vertices).map(|i| {
25        let angle_rad = 2. * PI / num_vertices as f64 * i as f64;
26        let radius = if i % 2 == 0 { r1 } else { r2 };
27        (angle_rad, radius)
28    });
29
30    let mut outer_points = Vec::new();
31    let mut inner_points = Vec::new();
32
33    for (angle_rad, radius) in vertex_iter {
34        let (sin, cos) = angle_rad.sin_cos();
35
36        let x = cos * radius;
37        let y = sin * radius;
38
39        outer_points.push([x, y]);
40        inner_points.push([x / 2., y / 2.]);
41    }
42
43    let bottom_surface = core.layers.topology.surfaces.xy_plane();
44    let sweep_path = Vector::from([0., 0., -h]);
45
46    Sketch::empty(&core.layers.topology)
47        .add_regions(
48            [Region::polygon(
49                outer_points,
50                core.layers.topology.surfaces.space_2d(),
51                core,
52            )
53            .add_interiors(
54                [Cycle::polygon(
55                    inner_points,
56                    core.layers.topology.surfaces.space_2d(),
57                    core,
58                )
59                .reverse(core)],
60                core,
61            )],
62            core,
63        )
64        .sweep_sketch(bottom_surface, sweep_path, core)
65}
