1use std::ops;
2
3use super::{Aabb, LineSegment, Point, Triangle, Vector};
4
5/// An affine transform
6#[repr(C)]
7#[derive(Debug, Clone, Copy, Default)]
8pub struct Transform {
9    /// # The [`nalgebra`] transform that does the actual work
10    pub inner: nalgebra::Transform<f64, nalgebra::TProjective, 3>,
11}
12
13impl Transform {
14    /// Construct an identity transform
15    pub fn identity() -> Self {
16        Self {
17            inner: nalgebra::Transform::identity(),
18        }
19    }
20
21    /// Construct a translation
22    pub fn translation(offset: impl Into<Vector<3>>) -> Self {
23        let offset = offset.into();
24
25        Self {
26            inner: nalgebra::Transform::from_matrix_unchecked(
27                nalgebra::OMatrix::new_translation(&offset.to_na()),
28            ),
29        }
30    }
31
32    /// Construct a rotation
33    ///
34    /// The direction of the vector defines the rotation axis. Its length
35    /// defines the angle of the rotation.
36    pub fn rotation(axis_angle: impl Into<Vector<3>>) -> Self {
37        let axis_angle = axis_angle.into();
38
39        Self {
40            inner: nalgebra::Transform::from_matrix_unchecked(
41                nalgebra::OMatrix::<_, nalgebra::Const<4>, _>::new_rotation(
42                    axis_angle.to_na(),
43                ),
44            ),
45        }
46    }
47
48    /// Construct a scaling
49    pub fn scale(s: impl Into<Vector<3>>) -> Self {
50        Self {
51            inner: nalgebra::Transform::from_matrix_unchecked(
52                nalgebra::OMatrix::new_nonuniform_scaling(&s.into().to_na()),
53            ),
54        }
55    }
56
57    /// # Extract the "right" vector from the rotational component
58    pub fn right(&self) -> Vector<3> {
59        let d = self.data();
60        Vector::from([d[0], d[1], d[2]])
61    }
62
63    /// # Extract the "up" vector from the rotational component
64    pub fn up(&self) -> Vector<3> {
65        let d = self.data();
66        Vector::from([d[4], d[5], d[6]])
67    }
68
69    /// Transform the given point
70    pub fn transform_point(&self, point: &Point<3>) -> Point<3> {
71        Point::from(self.inner.transform_point(&point.to_na()))
72    }
73
74    /// Inverse transform given point
75    pub fn inverse_transform_point(&self, point: &Point<3>) -> Point<3> {
76        Point::from(self.inner.inverse_transform_point(&point.to_na()))
77    }
78
79    /// Transform the given vector
80    pub fn transform_vector(&self, vector: &Vector<3>) -> Vector<3> {
81        Vector::from(self.inner.transform_vector(&vector.to_na()))
82    }
83
84    /// Transform the given segment
85    pub fn transform_segment(
86        &self,
87        segment: &LineSegment<3>,
88    ) -> LineSegment<3> {
89        let [a, b] = &segment.points;
90        LineSegment::from([self.transform_point(a), self.transform_point(b)])
91    }
92
93    /// Transform the given triangle
94    pub fn transform_triangle(&self, triangle: &Triangle<3>) -> Triangle<3> {
95        let [a, b, c] = &triangle.points;
96        Triangle::from([
97            self.transform_point(a),
98            self.transform_point(b),
99            self.transform_point(c),
100        ])
101    }
102
103    /// Inverse transform
104    pub fn inverse(&self) -> Self {
105        Self {
106            inner: self.inner.inverse(),
107        }
108    }
109
110    /// Transpose transform
111    pub fn transpose(&self) -> Self {
112        Self {
113            inner: nalgebra::Transform::from_matrix_unchecked(
114                self.inner.to_homogeneous().transpose(),
115            ),
116        }
117    }
118
119    /// Transform the given axis-aligned bounding box
120    pub fn transform_aabb(&self, aabb: &Aabb<3>) -> Aabb<3> {
121        Aabb {
122            min: self.transform_point(&aabb.min),
123            max: self.transform_point(&aabb.max),
124        }
125    }
126
127    /// Exposes the data of this Transform as a slice of f64.
128    pub fn data(&self) -> &[f64] {
129        self.inner.matrix().data.as_slice()
130    }
131
132    /// Extract the rotation component of this transform
133    pub fn extract_rotation(&self) -> Self {
134        Self {
135            inner: nalgebra::Transform::from_matrix_unchecked(
136                self.inner
137                    .matrix()
138                    .fixed_resize::<3, 3>(0.)
139                    .to_homogeneous(),
140            ),
141        }
142    }
143
144    /// Extract the translation component of this transform
145    pub fn extract_translation(&self) -> Self {
146        *self * self.extract_rotation().inverse()
147    }
148}
149
150impl ops::Mul<Self> for Transform {
151    type Output = Self;
152
153    fn mul(self, rhs: Self) -> Self::Output {
154        Self {
155            inner: self.inner.mul(rhs.inner),
156        }
157    }
158}

160#[cfg(test)]
161mod tests {
162    use approx::assert_abs_diff_eq;
163
164    use crate::{Scalar, Vector};
165
166    use super::Transform;
167
168    #[test]
169    fn extract_rotation_translation() {
170        let rotation =
171            Transform::rotation(Vector::unit_z() * (Scalar::PI / 2.));
172        let translation = Transform::translation([1., 2., 3.]);
173
174        assert_abs_diff_eq!(
175            (translation * rotation).extract_rotation().data(),
176            rotation.data(),
177            epsilon = 1e-8,
178        );
179
180        assert_abs_diff_eq!(
181            (translation * rotation).extract_translation().data(),
182            translation.data(),
183            epsilon = 1e-8,
184        );
185
186        assert_abs_diff_eq!(
187            (rotation * translation).extract_rotation().data(),
188            rotation.data(),
189            epsilon = 1e-8,
190        );
191
192        assert_abs_diff_eq!(
193            (rotation * translation).extract_translation().data(),
194            Transform::translation([-2., 1., 3.]).data(),
195            epsilon = 1e-8,
196        );
197    }
