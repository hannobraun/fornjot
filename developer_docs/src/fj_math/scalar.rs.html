12/// # A rational, finite scalar value
13///
14/// `Scalar` is a wrapper around `f64` which guarantees that the contained value
15/// is both rational and finite. This allows `Scalar` to provide
16/// implementations of [`Eq`], [`Ord`], and [`Hash`]; enabling `Scalar` (and
17/// types built on top of it), to be used as keys in various sets and maps.
18#[derive(Clone, Copy, Default)]
19#[repr(transparent)]
20pub struct Scalar {
21    value: f64,
22}

46    /// # Construct a `Scalar` from an `f64`
47    ///
48    /// ## Panics
49    ///
50    /// Panics, if the value provided is `NaN` or infinite.
51    pub fn from_f64(value: f64) -> Self {
52        if value.is_nan() {
53            panic!("`Scalar` value must not be `NaN`");
54        }
55        if value.is_infinite() {
56            panic!("`Scalar` value must not be infinite. Value: `{value}`");
57        }
58
59        Self { value }
60    }

78    pub fn is_negative(self) -> bool {
79        self < Self::ZERO

83    pub fn is_positive(self) -> bool {
84        self > Self::ZERO

88    pub fn is_zero(self) -> bool {
89        self == Self::ZERO

92    /// # Compute the sign of the scalar
93    pub fn sign(self) -> Sign {
94        if self.is_negative() {
95            return Sign::Negative;
96        }
97        if self.is_positive() {
98            return Sign::Positive;
99        }
100        if self.is_zero() {
101            return Sign::Zero;
102        }
103
104        unreachable!("Sign is neither negative, nor positive, nor zero.")
105    }

198impl Ord for Scalar {
199    fn cmp(&self, other: &Self) -> cmp::Ordering {
200        let Some(ordering) = self.value.partial_cmp(&other.value) else {
201            unreachable!(
202                "`Scalar` is not valid, but this has been checked by the \
203                constructor."
204            );
205        };
206
207        ordering
208    }

640/// # The sign of a [`Scalar`]
641///
642/// See [`Scalar::sign`]
643#[derive(Clone, Copy, Eq, PartialEq, Hash, Ord, PartialOrd)]
644pub enum Sign {
645    /// # The scalar value is negative
646    Negative,
647
648    /// # The scalar value is positive
649    Positive,
650
651    /// # The scalar value is zero
652    Zero,
653}
654
655impl Sign {
656    /// # Indicate whether the sign is negative
657    pub fn is_negative(&self) -> bool {
658        matches!(self, Self::Negative)
659    }
660
661    /// # Indicate whether the sign is positive
662    pub fn is_positive(&self) -> bool {
663        matches!(self, Self::Positive)
664    }
665
666    /// # Indicate whether the sign is zero
667    pub fn is_zero(&self) -> bool {
668        matches!(self, Self::Zero)
669    }
670
671    /// # Convert this sign back to a scalar
672    ///
673    /// Returns `-Scalar::ONE`, if the sign is negative; `Scalar::ONE`, if it is
674    /// positive; or `Scalar::ZERO`, if the value is zero.
675    pub fn to_scalar(self) -> Scalar {
676        match self {
677            Self::Negative => -Scalar::ONE,
678            Self::Positive => Scalar::ONE,
679            Self::Zero => Scalar::ZERO,
680        }
681    }
