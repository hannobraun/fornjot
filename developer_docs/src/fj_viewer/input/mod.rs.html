13/// # Tuning configuration for camera input
14pub struct CameraTuningConfig {
15    /// # Sensitivity of camera zoom, given scroll wheel input in lines
16    ///
17    /// Given a specific input, smaller values mean that the camera moves less,
18    /// larger values mean it moves more.
19    pub zoom_sensitivity_line: f64,
20
21    /// # Sensitivity of camera zoom, given scroll wheel input in pixels
22    ///
23    /// Given a specific input, smaller values mean that the camera moves less,
24    /// larger values mean it moves more.
25    pub zoom_sensitivity_pixel: f64,
26
27    /// # Sensitivity of camera rotation
28    ///
29    /// Given a specific input, smaller values mean that the camera rotates less,
30    /// larger values mean it rotates more.
31    pub rotation_sensitivity: f64,
32}
33
34/// # The default camera tuning configuration
35pub const DEFAULT_CAMERA_TUNING_CONFIG: CameraTuningConfig =
36    CameraTuningConfig {
37        zoom_sensitivity_line: 0.075,
38        zoom_sensitivity_pixel: 0.005,
39        rotation_sensitivity: 5.,
40    };
