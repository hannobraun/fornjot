17const SCALE_FACTOR: f64 = 0.13;
18const CUBE_TRANSLATION: [f64; 3] = [0.8, 0.7, 0.0];

20impl NavigationCubeRenderer {
21    pub fn new(
22        device: &wgpu::Device,
23        queue: &wgpu::Queue,
24        config: &wgpu::SurfaceConfiguration,
25    ) -> Self {
26        let texture_bind_group_layout =
27            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
28                entries: &[
29                    wgpu::BindGroupLayoutEntry {
30                        binding: 0,
31                        visibility: wgpu::ShaderStages::FRAGMENT,
32                        ty: wgpu::BindingType::Texture {
33                            multisampled: false,
34                            view_dimension: wgpu::TextureViewDimension::D2,
35                            sample_type: wgpu::TextureSampleType::Float {
36                                filterable: true,
37                            },
38                        },
39                        count: None,
40                    },
41                    wgpu::BindGroupLayoutEntry {
42                        binding: 1,
43                        visibility: wgpu::ShaderStages::FRAGMENT,
44                        ty: wgpu::BindingType::Sampler(
45                            wgpu::SamplerBindingType::Filtering,
46                        ),
47                        count: None,
48                    },
49                ],
50                label: Some("texture_bind_group_layout"),
51            });
52
53        let mvp_matrix_buffer =
54            device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
55                label: Some("Model Matrix Buffer"),
56                contents: bytemuck::cast_slice(&[
57                    transform::Transform::identity(),
58                ]),
59                usage: wgpu::BufferUsages::UNIFORM
60                    | wgpu::BufferUsages::COPY_DST,
61            });
62        let mvp_matrix_bind_group_layout =
63            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
64                entries: &[wgpu::BindGroupLayoutEntry {
65                    binding: 0,
66                    visibility: wgpu::ShaderStages::VERTEX,
67                    ty: wgpu::BindingType::Buffer {
68                        ty: wgpu::BufferBindingType::Uniform,
69                        has_dynamic_offset: false,
70                        min_binding_size: None,
71                    },
72                    count: None,
73                }],
74                label: Some("mvp_matrix_group_layout"),
75            });
76        let mvp_matrix_bind_group =
77            device.create_bind_group(&wgpu::BindGroupDescriptor {
78                layout: &mvp_matrix_bind_group_layout,
79                entries: &[wgpu::BindGroupEntry {
80                    binding: 0,
81                    resource: mvp_matrix_buffer.as_entire_binding(),
82                }],
83                label: Some("mvp_matrix_bind_group"),
84            });
85
86        let shader =
87            device.create_shader_module(wgpu::ShaderModuleDescriptor {
88                label: Some("Shadow Display Shader"),
89                source: wgpu::ShaderSource::Wgsl(

94        let render_pipeline_layout =
95            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
96                label: Some("Render Pipeline Layout"),
97                bind_group_layouts: &[
98                    &texture_bind_group_layout,
99                    &mvp_matrix_bind_group_layout,
100                ],
101                push_constant_ranges: &[],
102            });
103
104        let render_pipeline =
105            device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
106                label: Some("Navigation Cube Renderer"),
107                layout: Some(&render_pipeline_layout),
108                vertex: wgpu::VertexState {
109                    module: &shader,
110                    entry_point: Some("vertex"),
111                    compilation_options:
112                        wgpu::PipelineCompilationOptions::default(),
113                    buffers: &[model::ModelVertex::desc()],
114                },
115                fragment: Some(wgpu::FragmentState {
116                    module: &shader,
117                    entry_point: Some("fragment"),
118                    compilation_options:
119                        wgpu::PipelineCompilationOptions::default(),
120                    targets: &[Some(wgpu::ColorTargetState {
121                        format: config.format,
122                        blend: Some(wgpu::BlendState {
123                            color: wgpu::BlendComponent::REPLACE,
124                            alpha: wgpu::BlendComponent::REPLACE,
125                        }),
126                        write_mask: wgpu::ColorWrites::ALL,
127                    })],
128                }),
129                primitive: wgpu::PrimitiveState {
130                    topology: wgpu::PrimitiveTopology::TriangleList,
131                    strip_index_format: None,
132                    front_face: wgpu::FrontFace::Ccw,
133                    cull_mode: Some(wgpu::Face::Back),
134                    // Setting this to anything other than Fill requires Features::POLYGON_MODE_LINE
135                    // or Features::POLYGON_MODE_POINT
136                    polygon_mode: wgpu::PolygonMode::Fill,
137                    // Requires Features::DEPTH_CLIP_CONTROL
138                    unclipped_depth: false,
139                    // Requires Features::CONSERVATIVE_RASTERIZATION
140                    conservative: false,
141                },
142                depth_stencil: None,
143                multisample: wgpu::MultisampleState {
144                    count: 1,
145                    mask: !0,
146                    alpha_to_coverage_enabled: false,
147                },
148                multiview: None,
149                cache: None,
150            });
151
152        let cube_model =
153            load_model("cube.obj", device, queue, &texture_bind_group_layout)
154                .unwrap();
155
156        Self {
157            cube_model,
158            render_pipeline,
159            mvp_matrix_bind_group,
160            mvp_matrix_buffer,
161        }
162    }

164    pub fn draw(
165        &mut self,
166        view: &wgpu::TextureView,
167        encoder: &mut wgpu::CommandEncoder,
168        queue: &wgpu::Queue,
169        aspect_ratio: f64,
170        rotation: Transform,
171    ) {
172        let mvp_matrix = Self::get_mvp_matrix(rotation, aspect_ratio);
173        queue.write_buffer(
174            &self.mvp_matrix_buffer,
175            0,
176            bytemuck::cast_slice(&[mvp_matrix]),
177        );
178
179        let mut render_pass =
180            encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
181                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
182                    view,
183                    depth_slice: None,
184                    resolve_target: None,
185                    ops: wgpu::Operations {
186                        load: wgpu::LoadOp::Load,
187                        store: wgpu::StoreOp::Store,
188                    },
189                })],
190                ..Default::default()
191            });
192        render_pass.set_pipeline(&self.render_pipeline);
193        render_pass.set_bind_group(1, &self.mvp_matrix_bind_group, &[]);
194        render_pass.draw_model(&self.cube_model);
195    }
196
197    fn get_mvp_matrix(rotation: Transform, aspect_ratio: f64) -> [f32; 16] {
198        let scale = Transform::scale([SCALE_FACTOR; 3]);
199        let world_translation = Transform::translation([0.0, 0.0, -1.0]);
200
201        let mut model_matrix = Transform::identity();
202        model_matrix = model_matrix * world_translation;
203        model_matrix = model_matrix * rotation;
204        model_matrix = model_matrix * scale;
205
206        let perspective =
207            nalgebra::Perspective3::new(aspect_ratio, 30.0, 0.1, 2.0);
208
209        let view_matrix = nalgebra::Matrix4::look_at_lh(
210            &nalgebra::Point3::new(0.0, 0.0, 0.0),
211            &nalgebra::Point3::new(0.0, 0.0, 1.0),
212            &nalgebra::Vector3::new(0.0, -1.0, 0.0),
213        );
214
215        let screen_translation = Transform::translation(CUBE_TRANSLATION);
216
217        let matrix = screen_translation.inner.matrix()
218            * *perspective.to_projective().matrix()
219            * view_matrix
220            * model_matrix.inner.matrix();
221
222        let mut mat = [0.; 16];
223        mat.copy_from_slice(matrix.as_slice());
224        mat.map(|x| x as f32)
225    }
