16impl ModelVertex {
17    pub fn desc<'a>() -> wgpu::VertexBufferLayout<'a> {
18        use std::mem;
19        wgpu::VertexBufferLayout {
20            array_stride: mem::size_of::<ModelVertex>() as wgpu::BufferAddress,
21            step_mode: wgpu::VertexStepMode::Vertex,
22            attributes: &[
23                wgpu::VertexAttribute {
24                    offset: 0,
25                    shader_location: 0,
26                    format: wgpu::VertexFormat::Float32x3,

28                wgpu::VertexAttribute {
29                    offset: mem::size_of::<[f32; 3]>() as wgpu::BufferAddress,
30                    shader_location: 1,

33            ],

66pub fn load_model(
67    file_name: &str,
68    device: &wgpu::Device,
69    queue: &wgpu::Queue,
70    layout: &wgpu::BindGroupLayout,
71) -> Result<Model, LoadModelError> {
72    let assets = Assets::get_instance();
73
74    let (models, obj_materials) = tobj::load_obj_buf(
75        &mut assets.get_asset(file_name),
76        &tobj::LoadOptions {
77            triangulate: true,
78            single_index: true,
79            ..Default::default()
80        },
81        |p| {
82            tobj::load_mtl_buf(
83                &mut assets.get_asset(
84                    p.file_name()
85                        .unwrap()
86                        .to_str()
87                        .expect("OsStr could not be converted to a str"),
88                ),
89            )
90        },
91    )?;
92
93    let mut materials = Vec::new();
94    for m in obj_materials? {
95        let texture_data: &[u8] = match m.diffuse_texture {
96            Some(diffuse_texture) => assets.get_asset(diffuse_texture.as_str()),
97            None => continue,
98        };
99
100        let diffuse_texture = texture::Texture::from_bytes(
101            device,
102            queue,
103            texture_data,
104            file_name,
105        )?;
106
107        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
108            layout,
109            entries: &[
110                wgpu::BindGroupEntry {
111                    binding: 0,
112                    resource: wgpu::BindingResource::TextureView(
113                        &diffuse_texture.view,
114                    ),
115                },
116                wgpu::BindGroupEntry {
117                    binding: 1,
118                    resource: wgpu::BindingResource::Sampler(
119                        &diffuse_texture.sampler,
120                    ),
121                },
122            ],
123            label: None,
124        });
125
126        materials.push(Material { bind_group })
127    }
128
129    let meshes = models
130        .into_iter()
131        .map(|m| {
132            let vertices = (0..m.mesh.positions.len() / 3)
133                .map(|i| ModelVertex {
134                    position: [
135                        m.mesh.positions[i * 3],
136                        m.mesh.positions[i * 3 + 1],
137                        m.mesh.positions[i * 3 + 2],
138                    ],
139                    tex_coords: [
140                        m.mesh.texcoords[i * 2],
141                        1.0 - m.mesh.texcoords[i * 2 + 1],
142                    ],
143                })
144                .collect::<Vec<_>>();
145
146            let vertex_buffer =
147                device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
148                    label: Some(&format!("{file_name:?} Vertex Buffer")),
149                    contents: bytemuck::cast_slice(&vertices),
150                    usage: wgpu::BufferUsages::VERTEX,
151                });
152            let index_buffer =
153                device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
154                    label: Some(&format!("{file_name:?} Index Buffer")),
155                    contents: bytemuck::cast_slice(&m.mesh.indices),
156                    usage: wgpu::BufferUsages::INDEX,
157                });
158
159            Mesh {
160                vertex_buffer,
161                index_buffer,
162                num_elements: m.mesh.indices.len() as u32,
163                material: m.mesh.material_id.unwrap_or(0),
164            }
165        })
166        .collect::<Vec<_>>();
167
168    Ok(Model { meshes, materials })
169}

171pub trait DrawModel<'a> {
172    fn draw_mesh_instanced(
173        &mut self,
174        mesh: &'a Mesh,
175        material: &'a Material,
176        instances: Range<u32>,
177    );
178
179    fn draw_model(&mut self, model: &'a Model);
180    fn draw_model_instanced(&mut self, model: &'a Model, instances: Range<u32>);
181}
182
183impl<'a, 'b> DrawModel<'b> for wgpu::RenderPass<'a>
184where
185    'b: 'a,
186{
187    fn draw_mesh_instanced(
188        &mut self,
189        mesh: &'b Mesh,
190        material: &'b Material,
191        instances: Range<u32>,
192    ) {
193        self.set_vertex_buffer(0, mesh.vertex_buffer.slice(..));
194        self.set_index_buffer(
195            mesh.index_buffer.slice(..),
196            wgpu::IndexFormat::Uint32,
197        );
198        self.set_bind_group(0, &material.bind_group, &[]);
199        self.draw_indexed(0..mesh.num_elements, 0, instances);
200    }
201
202    fn draw_model(&mut self, model: &'b Model) {
203        self.draw_model_instanced(model, 0..1);
204    }
205
206    fn draw_model_instanced(
207        &mut self,
208        model: &'b Model,
209        instances: Range<u32>,
210    ) {
211        for mesh in &model.meshes {
212            let material = &model.materials[mesh.material];
213            self.draw_mesh_instanced(mesh, material, instances.clone());
214        }
215    }
216}
