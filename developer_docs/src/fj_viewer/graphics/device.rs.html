4pub struct Device {

9impl Device {
10    pub async fn from_preferred_adapter(
11        instance: &wgpu::Instance,
12        surface: &wgpu::Surface<'_>,
13    ) -> Result<(Self, wgpu::Adapter, wgpu::Features), DeviceError> {
14        let adapter = instance
15            .request_adapter(&wgpu::RequestAdapterOptions {
16                power_preference: wgpu::PowerPreference::None,
17                force_fallback_adapter: false,
18                compatible_surface: Some(surface),
19            })
20            .await?;
21
22        debug!("Using adapter: {:?}", adapter.get_info());
23
24        let (device, features) = Device::new(&adapter).await?;
25
26        Ok((device, adapter, features))
27    }
28
29    pub async fn try_from_all_adapters(
30        instance: &wgpu::Instance,
31    ) -> Result<(Self, wgpu::Adapter, wgpu::Features), DeviceError> {
32        #[cfg(not(target_arch = "wasm32"))]
33        {
34            let mut all_adapters = instance
35                .enumerate_adapters(wgpu::Backends::all())
36                .into_iter();
37
38            let result = loop {
39                let Some(adapter) = all_adapters.next() else {
40                    debug!("No more adapters to try");
41                    break None;
42                };
43
44                let (device, features) = match Device::new(&adapter).await {
45                    Ok((device, adapter)) => (device, adapter),
46                    Err(err) => {
47                        error!(
48                            "Failed to get device from adapter {:?}: {:?}",
49                            adapter.get_info(),
50                            err,
51                        );
52                        continue;
53                    }
54                };
55
56                break Some((device, adapter, features));
57            };
58
59            for adapter in all_adapters {
60                debug!(
61                    "Remaining adapter that wasn't tried: {:?}",
62                    adapter.get_info()
63                );
64            }
65
66            result.ok_or(DeviceError::FoundNoWorkingAdapter)
67        }
68        #[cfg(target_arch = "wasm32")]
69        {
70            _ = instance;
71            Err(DeviceError::FoundNoWorkingAdapter)
72        }
73    }
74
75    pub async fn new(
76        adapter: &wgpu::Adapter,
77    ) -> Result<(Self, wgpu::Features), DeviceError> {
78        let required_features = {
79            let desired_features = wgpu::Features::POLYGON_MODE_LINE;
80            let available_features = adapter.features();
81
82            // By requesting the intersection of desired and available features,
83            // we prevent two things:
84            //
85            // 1. That requesting the device panics, which would happen if we
86            //    requested unavailable features.
87            // 2. That a developer ends up accidentally using features that
88            //    happen to be available on their machine, but that aren't
89            //    necessarily available for all the users.
90            desired_features.intersection(available_features)
91        };
92
93        let required_limits = {
94            // This is the lowest of the available defaults. It should guarantee
95            // that we can run pretty much everywhere.
96            let lowest_limits = wgpu::Limits::downlevel_webgl2_defaults();
97
98            // However, these lowest limits aren't necessarily capable of
99            // supporting the screen resolution of our current platform, so
100            // let's amend them.
101            let supported_limits = adapter.limits();
102            lowest_limits.using_resolution(supported_limits)
103        };
104
105        let (device, queue) = adapter
106            .request_device(&wgpu::DeviceDescriptor {
107                label: None,
108                required_features,
109                required_limits,
110                experimental_features: wgpu::ExperimentalFeatures::disabled(),
111
112                // Here we give a memory hint to preserve memory usage. This
113                // should allow us to run in as much devices as possible.
114                memory_hints: wgpu::MemoryHints::MemoryUsage,
115
116                trace: wgpu::Trace::Off,
117            })
118            .await?;
119
120        Ok((Device { device, queue }, required_features))
121    }
122}

124/// Render device initialization error
125#[derive(Debug, thiserror::Error)]
126pub enum DeviceError {
127    /// Failed to request adapter
128    #[error("Failed to request adapter")]
129    RequestAdapter(#[from] wgpu::RequestAdapterError),
130
131    /// Failed to request device
132    #[error("Failed to request device")]
133    RequestDevice(#[from] wgpu::RequestDeviceError),
134
135    /// Found no working adapter to get a device from
136    #[error("Found no working adapter to get a device from")]
137    FoundNoWorkingAdapter,
138}
