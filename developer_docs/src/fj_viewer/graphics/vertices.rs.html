1use bytemuck::{Pod, Zeroable};
2use fj_interop::{Index, TriMesh, vertices_to_indexed_vertices};
3use fj_math::{Point, Scalar};
4
5#[derive(Debug)]
6pub struct Vertices {
7    vertices: Vec<Vertex>,
8    indices: Vec<Index>,
9}
10
11impl Vertices {
12    pub fn for_face(points: &[Point<2>]) -> Self {
13        let vertices = points
14            .iter()
15            .map(|point| {
16                let [x, y] = point.coords.components.map(Scalar::into_f32);
17
18                Vertex {
19                    position: [x, y, 0.],
20                    normal: [0., 0., 1.],
21                    color: [0., 0., 0., 1.],
22                }
23            })
24            .collect();
25        let indices = (0..).take(points.len()).collect();
26
27        Self { vertices, indices }
28    }
29
30    pub fn for_mesh(tri_mesh: &TriMesh) -> Self {
31        let (vertices, indices) = vertices_to_indexed_vertices(
32            tri_mesh.triangles.iter().flat_map(|triangle| {
33                let [a, b, c] = triangle.inner.points;
34
35                let normal = (b - a).cross(&(c - a)).normalize();
36                let color = triangle.color;
37
38                [a, b, c].map(|point| (point, normal, color))
39            }),
40            |(point, normal, color)| Vertex {
41                position: point.into(),
42                normal: normal.into(),
43                color: color.0.map(|v| f32::from(v) / 255.0),
44            },
45        );
46
47        Self { vertices, indices }
48    }
49
50    pub fn for_point(point: Point<3>) -> Self {
51        let vertices = vec![Vertex {
52            position: point.coords.components.map(|coord| coord.into_f32()),
53            normal: [0.; 3],
54            color: [0., 0., 0., 1.],
55        }];
56        let indices = vec![0];
57
58        Self { vertices, indices }
59    }
60
61    pub fn vertices(&self) -> &[Vertex] {
62        self.vertices.as_slice()
63    }
64
65    pub fn indices(&self) -> &[Index] {
66        self.indices.as_slice()
67    }
68}

71#[repr(C)]

75    pub color: [f32; 4],
