25impl GitHub {
26    /// Shells out to GitHub's CLI `gh` to try and determine if the commit belongs to any pull-request
27    pub fn find_pull_request_by(
28        sha: &str,
29        marker_label: &str,
30    ) -> anyhow::Result<Option<PullRequest>> {
31        log::trace!("listing pull-requests that contain the commit:");
32        let pulls = run_fun!(gh pr list --state merged --search ${sha} --limit 1 --json number)?;
33        log::trace!("{pulls}");
34
35        let pulls: Vec<PullRequest> = serde_json::from_str(&pulls)?;
36
37        if pulls.is_empty() {
38            log::debug!("the commit sha {sha} is not part of any pull-request");
39            return Ok(None);
40        }
41
42        let pr = pulls.first().unwrap().clone();
43        log::trace!("extracted {pr:?}");
44
45        log::trace!("getting labels for the possibly qualified pull-request:");
46        let labels = {
47            let pr_number = pr.number;
48            run_fun!(gh pr view ${pr_number} --json labels)?
49        };
50        log::trace!("{labels}");
51
52        let labels: Labels = serde_json::from_str(&labels)?;
53
54        if labels.labels.is_empty() {
55            log::debug!(
56            "the commit sha {sha} is not part of any pull-request with the {marker_label} label"
57        );
58            return Ok(None);
59        }
60
61        for label in labels.labels {
62            if label.name == marker_label {
63                return Ok(Some(pr));
64            }
65        }
66
67        Ok(None)
68    }
69}

74    // Set an "output" in GitHub Actions
75    pub fn set_output<'r>(
76        outputs: impl IntoIterator<Item = (Outputs, &'r str)>,
77    ) -> anyhow::Result<()> {
78        const GITHUB_OUTPUT: &str = "GITHUB_OUTPUT";
79
80        let output = env::var_os(GITHUB_OUTPUT).ok_or_else(|| {
81            anyhow!("Could not read environment variable {GITHUB_OUTPUT}")
82        })?;
83        let mut output = File::options().append(true).open(output)?;
84
85        for (key, value) in outputs {
86            log::debug!("setting output name={key} value={value}");
87            writeln!(output, "{key}={value}")?;
88        }
89
90        Ok(())
91    }
