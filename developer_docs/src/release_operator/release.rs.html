1use autolib::find_version_in_str;
2
3use crate::{Actions, GitHub};
4use std::fmt::{Display, Formatter};
5
6pub struct Release {
7    sha: String,
8    label: String,
9}
10
11pub enum Outputs {
12    ReleaseDetected,
13    TagName,
14}
15
16impl Display for Outputs {
17    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
18        match &self {
19            Self::ReleaseDetected => write!(f, "release-detected"),
20            Self::TagName => write!(f, "tag-name"),
21        }
22    }
23}
24
25impl Release {
26    pub fn new(sha: String, label: String) -> Self {
27        Self { sha, label }
28    }
29
30    pub fn detect(&self) -> anyhow::Result<()> {
31        let sha = &self.sha;
32        let label = &self.label;
33
34        // Try and find the pull-request that the commit was part of to examine it.
35        // A release can only ever be triggered by a pull-request being merged.
36        if GitHub::find_pull_request_by(sha, label)?.is_none() {
37            log::info!(
38                "Could not find a pull request with hash {sha} and label \
39                {label}",
40            );
41            return Self::miss();
42        }
43
44        let commit: String = cmd_lib::run_fun!(git log -n 1 "${sha}")?;
45
46        // A release commits need to contain a semver version number.
47        let version = find_version_in_str(&commit)?;
48
49        match version {
50            Some(v) => Self::hit(&v.to_string()),
51            None => {
52                log::info!(
53                    "Commit message is missing version number:\n\
54                    {commit}",
55                );
56                Self::miss()
57            }
58        }
59    }
60
61    fn hit(tag: &str) -> anyhow::Result<()> {
62        let tag = format!("v{tag}");
63        log::info!("detected release of {tag}");
64
65        Actions::set_output([
66            (Outputs::ReleaseDetected, "true"),
67            (Outputs::TagName, &tag),
68        ])?;
69
70        Ok(())
71    }
72
73    fn miss() -> anyhow::Result<()> {
74        log::info!("no release detected");
75        Actions::set_output([(Outputs::ReleaseDetected, "false")])?;
76        Ok(())
77    }
78}
