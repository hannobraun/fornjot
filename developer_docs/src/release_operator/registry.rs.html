30enum CrateState {
31    /// Our crate version is ahead of the registry and should be published
32    Ahead,
33    /// Our crate version is behind the registry; you'll be warned about this
34    Behind,
35    /// Our crate version matches the registry version
36    Published,
37    /// We encountered an unknown state while processing the crate
38    Unknown,
39}

49    pub fn publish_crates(&self) -> anyhow::Result<()> {
50        let mut crates = VecDeque::new();
51
52        for dir_entry in fs::read_dir("crates")? {
53            let dir_entry = dir_entry?;
54            let name = dir_entry
55                .file_name()
56                .into_string()
57                .map_err(|err| anyhow!("Error converting string: {err:?}"))?;
58
59            let mut command = MetadataCommand::new();
60            command.manifest_path(dir_entry.path().join("Cargo.toml"));
61
62            let metadata = command.exec()?;
63
64            crates.push_back((name, metadata));
65        }
66
67        let mut crates_ordered = Vec::new();
68        let mut processed_dependencies = BTreeSet::new();
69
70        while let Some((name, metadata)) = crates.pop_front() {
71            let mut package = None;
72            for p in &metadata.packages {
73                if p.name.as_str() == name {
74                    package = Some(p.clone());
75                }
76            }
77
78            let package = package.expect("Could not find package");
79
80            let mut unmet_dependencies = false;
81
82            for dependency in &package.dependencies {
83                let unmet_dependency = dependency.path.is_some()
84                    && !processed_dependencies.contains(&dependency.name);
85
86                if unmet_dependency {
87                    unmet_dependencies = true;
88                    break;
89                }
90            }
91
92            if unmet_dependencies {
93                crates.push_back((name, metadata));
94            } else {
95                let mut path = package.manifest_path.into_std_path_buf();
96                path.pop();
97
98                processed_dependencies.insert(name);
99                crates_ordered.push(Crate { path });
100            }
101        }
102
103        for c in crates_ordered {
104            c.validate()?;
105
106            match c.determine_state()? {
107                CrateState::Published | CrateState::Behind => continue,
108                CrateState::Unknown | CrateState::Ahead => {
109                    c.submit(&self.token, self.dry_run)?;
110                }
111            }
112        }
113
114        Ok(())
115    }
116}

118impl Crate {
119    fn validate(&self) -> anyhow::Result<()> {
120        match self.path.exists() {
121            true => Ok(()),
122            false => Err(anyhow!(
123                "given path to the '{self}' crate is either not readable or does not exist"
124            )),
125        }
126    }
127
128    fn get_local_version(&self) -> anyhow::Result<semver::Version> {
129        let name = format!("{self}");
130        let cargo_toml_location = std::fs::canonicalize(&self.path)
131            .context("absolute path to Cargo.toml")?;
132        let mut cmd = cargo_metadata::MetadataCommand::new();
133        cmd.manifest_path(format!(
134            "{}/Cargo.toml",
135            cargo_toml_location.to_string_lossy()
136        ))
137        .no_deps();
138
139        let metadata = cmd.exec()?;
140        let package = metadata
141            .packages
142            .iter()
143            .find(|p| p.name.as_str() == name)
144            .ok_or_else(|| anyhow!("could not find package"))?;
145
146        let version = package.version.to_owned();
147        log::debug!("{self} found as {version} on our side");
148
149        Ok(version)
150    }
151
152    fn get_upstream_version(&self) -> anyhow::Result<semver::Version> {
153        #[derive(Deserialize)]
154        struct CrateVersions {
155            versions: Vec<CrateVersion>,
156        }
157
158        #[derive(Deserialize)]
159        struct CrateVersion {
160            #[serde(rename = "num")]
161            version: semver::Version,
162        }
163
164        let client = reqwest::blocking::ClientBuilder::new()
165            .user_agent(concat!(
166                env!("CARGO_PKG_NAME"),
167                "/",
168                env!("CARGO_PKG_VERSION")
169            ))
170            .build()
171            .context("build http client")?;
172
173        let resp = client
174            .get(format!("https://crates.io/api/v1/crates/{self}"))
175            .send()
176            .context("fetching crate versions from the registry")?;
177
178        if resp.status() == reqwest::StatusCode::NOT_FOUND {
179            return Err(anyhow::Error::new(Error::CrateNotFound));
180        }
181
182        if resp.status() != reqwest::StatusCode::OK {
183            return Err(anyhow!(
184                "{self} request to crates.io failed with {} '{}'",
185                resp.status(),
186                resp.text().unwrap_or_else(|_| {
187                    "[response body could not be read]".to_string()
188                })
189            ));
190        }
191
192        let versions =
193            serde_json::from_str::<CrateVersions>(resp.text()?.as_str())
194                .context("deserializing crates.io response")?;
195
196        Ok(versions.versions.first().unwrap().version.to_owned())
197    }
198
199    fn determine_state(&self) -> anyhow::Result<CrateState> {
200        let theirs = match self.get_upstream_version() {
201            Ok(version) => version,
202            Err(error) => match error.downcast_ref::<Error>() {
203                Some(Error::CrateNotFound) => return Ok(CrateState::Unknown),
204                None => return Err(error),
205            },
206        };
207
208        let ours = self.get_local_version()?;
209
210        match theirs.cmp(&ours) {
211            std::cmp::Ordering::Less => Ok(CrateState::Ahead),
212            std::cmp::Ordering::Equal => {
213                log::info!("{self} has already been published as {ours}");
214                Ok(CrateState::Published)
215            }
216            std::cmp::Ordering::Greater => {
217                log::warn!("{self} has already been published as {ours}, which is a newer version");
218                Ok(CrateState::Behind)
219            }
220        }
221    }
222
223    fn submit(&self, token: &SecUtf8, dry_run: bool) -> anyhow::Result<()> {
224        log::info!("{self} publishing new version");
225
226        let ours = self.get_local_version()?;
227        let delay = Duration::from_secs(10);
228        let start_time = Instant::now();
229        let timeout = Duration::from_secs(600);
230        let mut attempt = 1;
231        let maximum_attempts = 10;
232
233        loop {
234            if Instant::now() - start_time > timeout {
235                bail!("{self} could not be published to the registry within {timeout:?}");
236            }
237
238            if attempt > maximum_attempts {
239                bail!("{self} could not be published to the registry; tried {attempt} times");
240            }
241
242            let mut command = Command::new("cargo");
243            command
244                .arg("publish")
245                .args(["--token", token.unsecure()])
246                .current_dir(&self.path);
247
248            if dry_run {
249                command.arg("--dry-run");
250            }
251
252            let exit_status = command.status()?;
253
254            if !exit_status.success() {
255                match exit_status.code() {
256                    Some(code) => {
257                        log::warn!("`cargo publish` failed with exit code {code}; trying again in {delay:?}");
258                        sleep(delay);
259                        attempt += 1;
260                        continue;
261                    }
262                    None => {
263                        bail!("`cargo publish` was terminated by signal")
264                    }
265                }
266            }
267
268            log::info!("{self} published as {ours}");
269            break;
270        }
271
272        Ok(())
273    }
