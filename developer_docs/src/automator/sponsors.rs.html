14    pub async fn query(octocrab: &Octocrab) -> anyhow::Result<Self> {
15        let graphql_query = "\
16            query {
17                viewer {
18                    sponsors(first: 100) {
19                        nodes {
20                            __typename
21                            ... on User {
22                                login
23                                sponsorshipForViewerAsSponsorable {
24                                    privacyLevel
25                                    createdAt
26                                    tier {
27                                        monthlyPriceInDollars
28                                    }
29                                    isOneTimePayment
30                                }
31                            }
32                            ... on Organization {
33                                login
34                                sponsorshipForViewerAsSponsorable {
35                                    privacyLevel
36                                    createdAt
37                                    tier {
38                                        monthlyPriceInDollars
39                                    }
40                                    isOneTimePayment
41                                }
42                            }
43                        }
44                    }
45                }
46            }";
47
48        let mut json_object = HashMap::new();
49        json_object.insert("query", graphql_query);
50
51        let response: serde_json::Value = octocrab
52            .graphql(&json_object)
53            .await
54            .context("GraphQL query failed")?;
55
56        debug!("Response to GraphQL query for sponsors:\n{response}");
57
58        let response: QueryResult = serde_json::from_value(response)
59            .context("Failed to deserialize GraphQL query result")?;
60
61        let mut sponsors = response
62            .data
63            .viewer
64            .sponsors
65            .nodes
66            .into_iter()
67            .filter_map(|node| {
68                if let QueryResultSponsorshipPrivacy::Private =
69                    node.sponsorship_for_viewer_as_sponsorable.privacy_level
70                {
71                    return None;
72                }
73
74                if node
75                    .sponsorship_for_viewer_as_sponsorable
76                    .is_one_time_payment
77                {
78                    return None;
79                }
80
81                let login = node.login;
82                let since =
83                    node.sponsorship_for_viewer_as_sponsorable.created_at;
84                let dollars = node
85                    .sponsorship_for_viewer_as_sponsorable
86                    .tier
87                    .monthly_price_in_dollars;
88
89                Some(Sponsor {
90                    login,
91                    since,
92                    dollars,
93                })
94            })
95            .collect::<Vec<_>>();
96
97        if sponsors.len() >= 100 {
98            todo!(
99                "Number of sponsors has reached max page size, but query does \
100                not support pagination."
101            )
102        }
103
104        sponsors.sort();
105
106        Ok(Self { inner: sponsors })
107    }

109    pub fn as_markdown(
110        &self,
111        min_dollars: u32,
112        for_readme: bool,
113    ) -> anyhow::Result<String> {
114        let mut output = String::from("Fornjot is supported by ");
115
116        for sponsor in &self.inner {
117            if sponsor.dollars < min_dollars {
118                continue;
119            }
120
121            let login = &sponsor.login;
122            let name = if for_readme {
123                format!("**@{login}**")
124            } else {
125                format!("@{login}")
126            };
127            let url = format!("https://github.com/{login}");
128
129            write!(output, "[{name}]({url}), ")?;
130        }
131
132        output.push_str(
133            "and [my other awesome sponsors](https://github.com/sponsors/hannobraun). Thank you!"
134        );
135
136        Ok(output)
137    }

147impl Ord for Sponsor {
148    fn cmp(&self, other: &Self) -> Ordering {
149        let by_dollars = other.dollars.cmp(&self.dollars);
150        let by_date = self.since.cmp(&other.since);
151        let by_login = self.login.cmp(&other.login);
152
153        if by_dollars.is_ne() {
154            return by_dollars;
155        }
156
157        if by_date.is_ne() {
158            return by_date;
159        }
160
161        by_login
162    }
163}
