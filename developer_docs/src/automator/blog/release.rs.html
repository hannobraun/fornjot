15pub async fn create_release_announcement(
16    octocrab: &Octocrab,
17) -> anyhow::Result<()> {
18    let date = util::now_ymd();
19
20    let pull_requests_since_last_release =
21        PullRequestsSinceLastRelease::fetch(octocrab)
22            .await
23            .context("Failed to fetch pull requests since last release")?;
24
25    let pull_requests =
26        pull_requests_since_last_release.pull_requests.into_values();
27
28    // For now, it's good enough to just release a new minor version every time.
29    // We could also determine whether there were breaking changes to make sure
30    // we actually need it, but as of now, breaking changes are pretty much a
31    // given.
32    let mut version = pull_requests_since_last_release.version_of_last_release;
33    version.minor += 1;
34    let version = version.to_string();
35
36    let min_dollars = 32;
37    let for_readme = false;
38    let sponsors = Sponsors::query(octocrab)
39        .await
40        .context("Failed to query sponsors")?
41        .as_markdown(min_dollars, for_readme)
42        .context("Failed to convert sponsor data to Markdown")?;
43
44    let mut file = util::create_blog_post_file("release", &version).await?;
45    generate_announcement(date, version, sponsors, pull_requests, &mut file)
46        .await
47        .context("Failed to generate release announcement")?;
48
49    Ok(())
50}

52async fn generate_announcement(
53    date: String,
54    version: String,
55    sponsors: String,
56    pull_requests: impl IntoIterator<Item = PullRequest>,
57    file: &mut File,
58) -> anyhow::Result<()> {
59    let mut pull_request_list = String::new();
60    let mut pull_request_links = String::new();
61    let mut author_links = String::new();
62
63    let author_blacklist = hash_set! {
64        "hannobraun",
65        "dependabot[bot]"
66    };
67    let mut authors = HashSet::new();
68
69    for pull_request in pull_requests {
70        let PullRequest {
71            number,
72            title,
73            url,
74            author,
75            ..
76        } = pull_request;
77
78        let author = if author_blacklist.contains(author.name.as_str()) {
79            None
80        } else {
81            Some(author)
82        };
83
84        let thanks = match author.as_ref() {
85            Some(author) => format!("; thank you, [@{}]!", author.name),
86            None => String::new(),
87        };
88
89        let item = format!("- {title} ([#{number}]{thanks})\n");
90        pull_request_list.push_str(&item);
91
92        let link = format!("[#{number}]: {url}\n");
93        pull_request_links.push_str(&link);
94
95        if let Some(Author { name, profile }) = author {
96            if !authors.contains(&name) {
97                let author_link = format!("[@{name}]: {profile}\n");
98                author_links.push_str(&author_link);
99
100                authors.insert(name.clone());
101            }
102        }
103    }
104
105    let mut buf = String::new();
106    write!(
107        buf,
108        "\
109+++
110title = \"Fornjot {version}\"
111# TASK: Uncomment this date, once the announcement is ready to be published.
112# date = {date}
113
114# Uncomment to generate the HTML for the email newsletter.
115# template = \"newsletter/release.html\"
116
117[extra]
118version = \"{version}\"
119# TASK: Choose a descriptive subtitle.
120subtitle = \"This is a subtitle\"
121+++
122
123**TASK: Write introduction.**
124
125
126### Sponsors
127
128{sponsors}
129
130<strong class=\"call-to-action\">
131    <p>
132        If you want Fornjot to be sustainable long-term, please consider <a href=\"https://github.com/sponsors/hannobraun\">supporting me</a> too.
133    </p>
134</strong>
135
136
137### Library improvements
138
139Improvements to Fornjot libraries.
140
141#### `fj-core`
142
143**TASK: Add library improvements.**
144
145
146### Other changes
147
148Improvements that are not associated with a specific Fornjot library.
149
150**TASK: Add internal improvements.**
151
152
153### Unsorted pull requests
154
155**TASK: Sort into the categories above; update/merge as appropriate.**
156
157{pull_request_list}
158{pull_request_links}
159{author_links}\
160    "
161    )?;
162
163    file.write_all(buf.as_bytes()).await?;
164
165    Ok(())
166}
