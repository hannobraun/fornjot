7pub fn now_ymd() -> String {
8    let now = Utc::now();
9    format!("{}-{:02}-{:02}", now.year(), now.month(), now.day())
10}

12pub async fn create_blog_post_file(
13    category: &str,
14    title: &str,
15) -> anyhow::Result<File> {
16    let dir = PathBuf::from(format!("content/blog/{category}/{title}"));
17    let file = dir.join("index.md");
18
19    // VS Code (and probably other editors/IDEs) renders the path in the output
20    // as a clickable link, so the user can open the file easily.
21    println!("Generating `{category}` blog post at {}", file.display());
22
23    fs::create_dir_all(&dir).await.with_context(|| {
24        format!("Failed to create directory `{}`", dir.display())
25    })?;
26    let file = File::create(&file).await.with_context(|| {
27        format!("Failed to create file `{}`", file.display())
28    })?;
29
30    Ok(file)
31}
