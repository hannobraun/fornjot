1//! Cross-compiler for the Fornjot build
2//!
3//! This tools cross-compiles the Fornjot crates that support that to the
4//! targets they support. This is less resource-intense than using a `matrix` in
5//! GitHub Actions (which would start one build job per crate and target), and
6//! allows for the cross-compilation code to be re-used in `justfile`.
7
8use std::process::Command;
9
10use anyhow::anyhow;
11
12fn main() -> anyhow::Result<()> {
13    let targets = [
14        Target {
15            triple: "aarch64-apple-ios",
16            crates: &["fj-core", "fj-export", "fj-interop", "fj-math"],
17        },
18        Target {
19            triple: "aarch64-linux-android",
20            crates: &["fj-core", "fj-export", "fj-interop", "fj-math"],
21        },
22        Target {
23            triple: "wasm32-unknown-unknown",
24            crates: &[
25                "fj-core",
26                "fj-export",
27                "fj-interop",
28                "fj-math",
29                "fj-viewer",
30            ],
31        },
32    ];
33
34    for target in targets {
35        for crate_ in target.crates {
36            let mut command = Command::new("cargo");
37            command
38                .arg("build")
39                .arg("--all-features")
40                .args(["--target", target.triple])
41                .args(["-p", crate_])
42                .env("RUSTFLAGS", "-D warnings");
43
44            println!("Running {command:?}");
45            let status = command.status()?;
46
47            if !status.success() {
48                return Err(anyhow!("Cargo exited with error code: {status}"));
49            }
50        }
51    }
52
53    Ok(())
54}
55
56struct Target {
57    /// The target triple
58    triple: &'static str,
59
60    /// The crates that are supported on this target
61    crates: &'static [&'static str],
62}
