7/// An ordered set of objects
8///
9/// This is the data structure used by all objects that reference multiple
10/// objects of the same type. It is a set, not containing any duplicate
11/// elements, and it maintains the insertion order of those elements.
12#[derive(Clone, Debug)]
13pub struct ObjectSet<T> {
14    // This is supposed to be a set data structure, so what is that `Vec` doing
15    // here? Well, it's here because we need it to preserve insertion order, but
16    // that doesn't explain why it is here *alone*.
17    //
18    // If you look closely, you'll notice that this is an immutable data
19    // structure (since it is used in objects, and objects themselves are
20    // immutable). We need to make sure there are no duplicates when this is
21    // constructed (see the constructor below), but after that, we're fine.
22    inner: Vec<Handle<T>>,
23}
24
25impl<T> ObjectSet<T> {
26    /// Create an instance of `ObjectSet` from an iterator over `Handle<T>`
27    ///
28    /// # Panics
29    ///
30    /// Panics, if the iterator contains duplicate `Handle`s.
31    pub fn new(handles: impl IntoIterator<Item = Handle<T>>) -> Self
32    where
33        T: Debug,
34    {
35        let mut added = BTreeSet::new();
36        let mut inner = Vec::new();
37
38        for handle in handles.into_iter() {
39            if added.contains(&handle) {
40                panic!(
41                    "Constructing `ObjectSet` with duplicate handle: {:?}",
42                    handle
43                );
44            }
45
46            added.insert(handle.clone());
47            inner.push(handle);
48        }
49
50        Self { inner }
51    }
52
53    /// Return the number of objects in this set
54    pub fn len(&self) -> usize {
55        self.inner.len()
56    }
57
58    /// Indicate whether the set is empty
59    pub fn is_empty(&self) -> bool {
60        self.inner.is_empty()
61    }
62
63    /// Indicate whether the set contains the provided object
64    pub fn contains(&self, object: &Handle<T>) -> bool {
65        self.index_of(object).is_some()
66    }
67
68    /// Return the only item
69    ///
70    /// # Panics
71    ///
72    /// Panics, if there is more than one item.
73    pub fn only(&self) -> &Handle<T> {
74        let mut iter = self.inner.iter();
75        let item = iter
76            .next()
77            .expect("Requested only item, but no items available");
78
79        assert!(
80            iter.next().is_none(),
81            "Requested only item, but more than one available"
82        );
83
84        item
85    }
86
87    /// Return the first item
88    ///
89    /// # Panics
90    ///
91    /// Panics, if there are no items.
92    pub fn first(&self) -> &Handle<T> {
93        self.inner
94            .first()
95            .expect("Requested first item, but no items available")
96    }
97
98    /// Return the n-th item
99    pub fn nth(&self, index: usize) -> Option<&Handle<T>> {
100        self.inner.get(index)
101    }
102
103    /// Return the n-th item, treating the index space as circular
104    ///
105    /// If the length of `ObjectSet` is `i`, then retrieving the i-th edge using
106    /// this method, is the same as retrieving the 0-th one, and so on.
107    ///
108    /// # Panics
109    ///
110    /// Panics, if `ObjectSet` is empty.
111    pub fn nth_circular(&self, index: usize) -> &Handle<T> {
112        assert!(!self.is_empty(), "`ObjectSet` must not be empty");
113
114        let index = index % self.len();
115        self.nth(index)
116            .expect("Index must be valid, due to modulo above")
117    }
118
119    /// Return the index of the item, if available
120    pub fn index_of(&self, handle: &Handle<T>) -> Option<usize> {
121        self.inner.iter().position(|h| h.id() == handle.id())
122    }
123
124    /// Access the item before the provided one
125    ///
126    /// Returns `None`, if the provided item is not in this set.
127    pub fn before(&self, handle: &Handle<T>) -> Option<&Handle<T>> {
128        self.index_of(handle)
129            .map(|index| self.nth_circular(index + self.len() - 1))
130    }
131
132    /// Access the item after the provided one
133    ///
134    /// Returns `None`, if the provided item is not in this set.
135    pub fn after(&self, handle: &Handle<T>) -> Option<&Handle<T>> {
136        self.index_of(handle)
137            .map(|index| self.nth_circular(index + 1))
138    }
139
140    /// Access an iterator over the objects
141    pub fn iter(&self) -> ObjectSetIter<'_, T> {
142        self.inner.iter()
143    }
144
145    /// Access an iterator over the neighboring pairs of all contained objects
146    pub fn pairs(&self) -> impl Iterator<Item = (&Handle<T>, &Handle<T>)> {
147        self.iter().circular_tuple_windows()
148    }
149
150    /// Create a new instance in which the provided object has been replaced
151    ///
152    /// Returns `None`, if the provided item is not present.
153    ///
154    /// # Panics
155    ///
156    /// Panics, if the update results in a duplicate item.
157    #[must_use]
158    pub fn replace(
159        &self,
160        original: &Handle<T>,
161        replacements: impl IntoIterator<Item = impl Into<Handle<T>>>,
162    ) -> Option<Self>
163    where
164        T: Debug,
165    {
166        let mut iter = self.iter().cloned().peekable();
167
168        // Collect all items before the item we want to update.
169        let mut before = Vec::new();
170        loop {
171            let next = match iter.next() {
172                Some(handle) => handle,
173                None => {
174                    // We went through the whole iterator without finding the
175                    // item we were looking for.
176                    return None;
177                }
178            };
179
180            if next.id() == original.id() {
181                break;
182            }
183
184            before.push(next.clone());
185        }
186
187        // What's left in the iterator is what comes after the replaced item.
188        // Let's make that a bit more explicit by renaming the variable.
189        let after = iter;
190
191        Some(
192            before
193                .into_iter()
194                .chain(replacements.into_iter().map(|handle| handle.into()))
195                .chain(after)
196                .collect(),
197        )
198    }
199}

259/// An owned iterator over an [`ObjectSet`]
260///
261/// You might wonder why we're returning references to `Handle`s here, when
262/// `Handle` already is a kind of reference, and easily cloned.
263///
264/// Most of the time that doesn't make a difference, but there are use cases
265/// where dealing with owned `Handle`s is inconvenient, for example when using
266/// iterator adapters. You can't return a reference to the argument of an
267/// adapter's closure, if you own that argument. You can, if you just reference
268/// the argument.
269pub type ObjectSetIntoIter<T> = vec::IntoIter<Handle<T>>;
