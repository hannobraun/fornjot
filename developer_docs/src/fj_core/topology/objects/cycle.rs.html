4    geometry::{Geometry, Path},

11/// A cycle of connected edges
12#[derive(Clone, Debug)]
13pub struct Cycle {
14    half_edges: ObjectSet<HalfEdge>,
15}
16
17impl Cycle {
18    /// Create an instance of `Cycle`
19    pub fn new(half_edges: impl IntoIterator<Item = Handle<HalfEdge>>) -> Self {
20        let half_edges = half_edges.into_iter().collect();
21        Self { half_edges }
22    }
23
24    /// Access the edges that make up the cycle
25    pub fn half_edges(&self) -> &ObjectSet<HalfEdge> {
26        &self.half_edges
27    }
28
29    /// Indicate the cycle's winding, assuming a right-handed coordinate system
30    ///
31    /// Please note that this is not *the* winding of the cycle, only one of the
32    /// two possible windings, depending on the direction you look at the
33    /// surface that the cycle is defined on from.
34    pub fn winding(
35        &self,
36        geometry: &Geometry,
37        surface: &Handle<Surface>,
38    ) -> Winding {
39        // The cycle could be made up of one or two circles. If that is the
40        // case, the winding of the cycle is determined by the winding of the
41        // first circle.
42        if self.half_edges.len() < 3 {
43            let first = self
44                .half_edges()
45                .iter()
46                .next()
47                .expect("Invalid cycle: expected at least one edge");
48
49            let curve_geom = geometry
50                .of_curve(first.curve())
51                .unwrap()
52                .local_on(surface)
53                .unwrap()
54                .clone();
55
56            let [a, b] = [
57                curve_geom.path.point_from_path_coords(
58                    geometry
59                        .of_vertex(first.start_vertex())
60                        .unwrap()
61                        .local_on(first.curve())
62                        .unwrap()
63                        .position,
64                ),
65                curve_geom.path.point_from_path_coords(
66                    geometry
67                        .of_vertex(
68                            self.half_edges()
69                                .after(first)
70                                .expect("Just got half-edge from this cycle")
71                                .start_vertex(),
72                        )
73                        .unwrap()
74                        .local_on(first.curve())
75                        .unwrap()
76                        .position,
77                ),
78            ];
79            let edge_direction_positive = a < b;
80
81            let circle = match curve_geom.path {
82                Path::Circle(circle) => circle,
83                Path::Line(_) => unreachable!(
84                    "Invalid cycle: less than 3 edges, but not all are circles"
85                ),
86            };
87            let cross_positive = circle.a().cross2d(&circle.b()) > Scalar::ZERO;
88
89            if edge_direction_positive == cross_positive {
90                return Winding::Ccw;
91            } else {
92                return Winding::Cw;
93            }
94        }
95
96        // Now that we got the special case out of the way, we can treat the
97        // cycle as a polygon:
98        // https://stackoverflow.com/a/1165943
99
100        let mut sum = Scalar::ZERO;
101
102        for (a, b) in self.half_edges().pairs() {
103            let [a, b] = [a, b].map(|half_edge| {
104                geometry
105                    .of_curve(half_edge.curve())
106                    .unwrap()
107                    .local_on(surface)
108                    .unwrap()
109                    .path
110                    .point_from_path_coords(
111                        geometry
112                            .of_vertex(half_edge.start_vertex())
113                            .unwrap()
114                            .local_on(half_edge.curve())
115                            .unwrap()
116                            .position,
117                    )
118            });
119
120            sum += (b.u - a.u) * (b.v + a.v);
121        }
122
123        if sum > Scalar::ZERO {
124            return Winding::Cw;
125        }
126        if sum < Scalar::ZERO {
127            return Winding::Ccw;
128        }
129
130        unreachable!("Encountered invalid cycle: {self:#?}");
131    }
132}
