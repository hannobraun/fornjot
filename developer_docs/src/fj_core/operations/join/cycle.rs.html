17/// Join a [`Cycle`] to another
18pub trait JoinCycle {
19    /// Add new half-edges to the cycle that are joined to the provided ones
20    ///
21    /// This method creates a new half-edge for each half-edge that is provided,
22    /// joins the new half-edge to the provided one, and adds the new half-edge
23    /// to the cycle.
24    ///
25    /// Expects the provided half-edges to be in an unnatural order:
26    ///
27    /// - This is the opposite order that they would appear in, in a cycle.
28    /// - Meaning each half-edge ends where the _previous_ one starts.
29    ///
30    /// The geometry for each new half-edge needs to be provided as well. Also
31    /// requires the surface that the cycle is defined in.
32    #[must_use]
33    fn add_joined_half_edges<Es>(
34        &self,
35        half_edges: Es,
36        surface: Handle<Surface>,
37        core: &mut Core,
38    ) -> Self
39    where
40        Es: IntoIterator<Item = (Handle<HalfEdge>, LocalCurveGeom)>,
41        Es::IntoIter: Clone + ExactSizeIterator;
42
43    /// Join the cycle to another
44    ///
45    /// Joins the cycle to the other at the provided ranges. The ranges specify
46    /// the indices of the edges that are joined together.
47    ///
48    /// A modulo operation is applied to all indices before use, so in a cycle
49    /// of 3 edges, indices `0` and `3` refer to the same edge. This allows for
50    /// specifying a range that crosses the "seam" of the cycle.
51    ///
52    /// # Panics
53    ///
54    /// Panics, if the ranges have different lengths.
55    ///
56    /// # Assumptions
57    ///
58    /// This method makes some assumptions that need to be met, if the operation
59    /// is to result in a valid shape:
60    ///
61    /// - **The joined edges must be coincident.**
62    /// - **The locally defined curve coordinate systems of the edges must
63    ///   match.**
64    ///
65    /// If either of those assumptions are not met, this will result in a
66    /// validation error down the line.
67    ///
68    /// # Implementation Note
69    ///
70    /// The use of the `RangeInclusive` type might be a bit limiting, as other
71    /// range types might be more convenient in a given use case. This
72    /// implementation was chosen for now, as it wasn't clear whether the
73    /// additional complexity of using `RangeBounds` would be worth it.
74    ///
75    /// A solution based on `SliceIndex` could theoretically be used, but that
76    /// trait is partially unstable. In addition, it's not clear how it could be
77    /// used generically, as it could yield a range (which can be iterated over)
78    /// or a single item (which can not). This is not a hard problem in
79    /// principle (a single item could just be an iterator of length 1), but I
80    /// don't see it how to address this in Rust in a reasonable way.
81    ///
82    /// Maybe a custom trait that is implemented for `usize` and all range types
83    /// would be the best solution.
84    #[must_use]
85    fn join_to(
86        &self,
87        other: &Cycle,
88        range: RangeInclusive<usize>,
89        other_range: RangeInclusive<usize>,
90        surface_self: Handle<Surface>,
91        core: &mut Core,
92    ) -> Self;
93}
94
95impl JoinCycle for Cycle {
96    fn add_joined_half_edges<Es>(
97        &self,
98        half_edges: Es,
99        surface: Handle<Surface>,
100        core: &mut Core,
101    ) -> Self
102    where
103        Es: IntoIterator<Item = (Handle<HalfEdge>, LocalCurveGeom)>,
104        Es::IntoIter: Clone + ExactSizeIterator,
105    {
106        let half_edges = half_edges
107            .into_iter()
108            .circular_tuple_windows()
109            // If you're confused about the naming of `next_half_edge` and/or
110            // the relative order of `half_edge`/`next_half_edge`, please refer
111            // to the documentation of this method. This matches the order it
112            // expects half-edges in.
113            .map(|((next_half_edge, _), (half_edge, curve_geom))| {
114                let half_edge = HalfEdge::unjoined(core)
115                    .update_curve(|_, _| half_edge.curve().clone(), core)
116                    .update_start_vertex(
117                        |_, _| next_half_edge.start_vertex().clone(),
118                        core,
119                    )
120                    .insert(core);
121
122                core.layers.geometry.define_curve(
123                    half_edge.curve().clone(),
124                    surface.clone(),
125                    curve_geom,
126                );
127
128                half_edge
129            })
130            .collect::<Vec<_>>();
131        self.add_half_edges(half_edges, core)
132    }
133
134    fn join_to(
135        &self,
136        other: &Cycle,
137        range: RangeInclusive<usize>,
138        range_other: RangeInclusive<usize>,
139        surface_self: Handle<Surface>,
140        core: &mut Core,
141    ) -> Self {
142        assert_eq!(
143            range.end() - range.start(),
144            range_other.end() - range_other.start(),
145            "Ranges have different lengths",
146        );
147
148        range.zip(range_other).fold(
149            self.clone(),
150            |cycle, (index, index_other)| {
151                let half_edge = self.half_edges().nth_circular(index);
152                let half_edge_next = self.half_edges().nth_circular(index + 1);
153                let half_edge_other =
154                    other.half_edges().nth_circular(index_other);
155                let half_edge_other_next =
156                    other.half_edges().nth_circular(index_other + 1);
157
158                cycle
159                    .update_half_edge(
160                        half_edge,
161                        |half_edge, core| {
162                            // The curve of the other half-edge we're joining
163                            // this one to already has a curve geometry,
164                            // presumably. But it might not have a local
165                            // definition for the surface that *this* half-edge
166                            // is in.
167                            //
168                            // We need to make sure that any local definition
169                            // that our current curve already has, moves over to
170                            // the new one.
171                            let curve_geom = core
172                                .layers
173                                .geometry
174                                .of_curve(half_edge.curve())
175                                .and_then(|curve_geom| {
176                                    curve_geom.local_on(&surface_self)
177                                });
178                            if let Some(curve_geom) = curve_geom {
179                                core.layers.geometry.define_curve(
180                                    half_edge_other.curve().clone(),
181                                    surface_self.clone(),
182                                    curve_geom.clone(),
183                                );
184                            }
185
186                            // The same goes for vertices. We have to move over
187                            // any local definitions we have to the other
188                            // vertices.
189                            let vertex_geom_prev_end = core
190                                .layers
191                                .geometry
192                                .of_vertex(half_edge.start_vertex())
193                                .and_then(|vertex_geom| {
194                                    vertex_geom.local_on(
195                                        self.half_edges()
196                                            .before(half_edge)
197                                            .unwrap()
198                                            .curve(),
199                                    )
200                                })
201                                .cloned();
202                            let vertex_geom_start = core
203                                .layers
204                                .geometry
205                                .of_vertex(half_edge.start_vertex())
206                                .and_then(|vertex_geom| {
207                                    vertex_geom.local_on(half_edge.curve())
208                                })
209                                .cloned();
210                            let vertex_geom_end = core
211                                .layers
212                                .geometry
213                                .of_vertex(half_edge_next.start_vertex())
214                                .and_then(|vertex_geom| {
215                                    vertex_geom.local_on(half_edge.curve())
216                                })
217                                .cloned();
218                            if let Some(vertex_geom) = vertex_geom_prev_end {
219                                core.layers.geometry.define_vertex(
220                                    half_edge_other_next.start_vertex().clone(),
221                                    self.half_edges()
222                                        .before(half_edge)
223                                        .unwrap()
224                                        .curve()
225                                        .clone(),
226                                    vertex_geom,
227                                );
228                            }
229                            if let Some(vertex_geom_start) = vertex_geom_start {
230                                core.layers.geometry.define_vertex(
231                                    half_edge_other_next.start_vertex().clone(),
232                                    half_edge_other.curve().clone(),
233                                    vertex_geom_start,
234                                );
235                            }
236                            if let Some(vertex_geom_end) = vertex_geom_end {
237                                core.layers.geometry.define_vertex(
238                                    half_edge_other.start_vertex().clone(),
239                                    half_edge_other.curve().clone(),
240                                    vertex_geom_end,
241                                );
242                            }
243
244                            [half_edge
245                                .update_curve(
246                                    |_, _| half_edge_other.curve().clone(),
247                                    core,
248                                )
249                                .update_start_vertex(
250                                    |_, _| {
251                                        half_edge_other_next
252                                            .start_vertex()
253                                            .clone()
254                                    },
255                                    core,
256                                )
257                                .insert(core)]
258                        },
259                        core,
260                    )
261                    .update_half_edge(
262                        half_edge_next,
263                        |half_edge, core| {
264                            // And we need to move over the geometry for this
265                            // vertex too.
266                            let vertex_geom = core
267                                .layers
268                                .geometry
269                                .of_vertex(half_edge_next.start_vertex())
270                                .and_then(|vertex_geom| {
271                                    vertex_geom.local_on(half_edge_next.curve())
272                                })
273                                .cloned();
274                            if let Some(vertex_geom) = vertex_geom {
275                                core.layers.geometry.define_vertex(
276                                    half_edge_other.start_vertex().clone(),
277                                    half_edge.curve().clone(),
278                                    vertex_geom,
279                                );
280                            }
281
282                            [half_edge
283                                .update_start_vertex(
284                                    |_, _| {
285                                        half_edge_other.start_vertex().clone()
286                                    },
287                                    core,
288                                )
289                                .insert(core)]
290                        },
291                        core,
292                    )
293            },
294        )
295    }
