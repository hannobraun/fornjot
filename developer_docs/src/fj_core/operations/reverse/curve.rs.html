1use crate::{
2    Core,

10impl ReverseCurveCoordinateSystems for (&Handle<Curve>, &Handle<Surface>) {
11    type Reversed = Handle<Curve>;
12
13    fn reverse_curve_coordinate_systems(
14        self,
15        core: &mut Core,
16    ) -> Self::Reversed {
17        // We don't *actually* need a surface here, because *all* local
18        // definitions need to be reversed in sync anyway. However, then this
19        // method couldn't be called in `HalfEdge`'s implementation, meaning the
20        // way this trait works had to change completely.
21        //
22        // At this point it's probably not worth it to re-architect the trait,
23        // since eventually, redundant geometry definitions shouldn't exist any
24        // more, and then the existence of this trait, at least in its current
25        // form, is in question anyway.
26        //
27        // Once the dust has settled, we can figure out what needs to happen
28        // here. In the meantime, we can have this weird implementation, and
29        // rely on the callers to make sure everything stays consistent.
30        //
31        // [1]: https://github.com/hannobraun/fornjot/issues/2290
32        let (curve, surface) = self;
33
34        let mut curve_geom = core
35            .layers
36            .geometry
37            .of_curve(curve)
38            .unwrap()
39            .local_on(surface)
40            .unwrap()
41            .clone();
42        curve_geom.path = curve_geom.path.reverse();
43
44        let curve = Curve::new().insert(core).derive_from(curve, core);
45
46        core.layers.geometry.define_curve(
47            curve.clone(),
48            surface.clone(),
49            curve_geom,
50        );
51
52        curve
53    }
54}
