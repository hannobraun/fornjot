12impl TransformObject for (&Handle<Curve>, &Handle<Surface>) {
13    type Transformed = Handle<Curve>;
14
15    fn transform_with_cache(
16        self,
17        transform: &Transform,
18        core: &mut Core,
19        cache: &mut TransformCache,
20    ) -> Self::Transformed {
21        let (curve, surface) = self;
22
23        // We don't actually need to transform the curve, as its geometry is
24        // locally defined on a surface. We need to transform that surface
25        // though, or what we do here will have no consequence.
26        //
27        // If this transformation is only one element in the transformation of
28        // a whole object graph, using the cache here ensures that the surface
29        // doesn't get transformed multiple times.
30        let transformed_surface = cache
31            .entry(surface)
32            .or_insert_with(|| surface.transform(transform, core))
33            .clone();
34        let transformed_curve = cache
35            .entry(curve)
36            .or_insert_with(|| Curve::new().insert(core))
37            .clone();
38
39        core.layers.geometry.define_curve(
40            transformed_curve.clone(),
41            transformed_surface,
42            core.layers
43                .geometry
44                .of_curve(curve)
45                .unwrap()
46                .local_on(surface)
47                .unwrap()
48                .clone(),
49        );
50
51        transformed_curve
52    }
53}
