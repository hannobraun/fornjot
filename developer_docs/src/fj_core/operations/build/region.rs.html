10/// Build a [`Region`]
11///
12/// See [module-level documentation] for context.
13///
14/// [module-level documentation]: super
15pub trait BuildRegion {
16    /// Build an empty region
17    fn empty(core: &mut Core) -> Region {
18        let exterior = Cycle::empty().insert(core);
19        let interiors = [];
20
21        Region::new(exterior, interiors)
22    }
23
24    /// Build a circle
25    fn circle(
26        center: impl Into<Point<2>>,
27        radius: impl Into<Scalar>,
28        surface: Handle<Surface>,
29        core: &mut Core,
30    ) -> Region {
31        let exterior =
32            Cycle::circle(center, radius, surface, core).insert(core);
33        Region::new(exterior, [])
34    }
35
36    /// Build a polygon
37    fn polygon<P, Ps>(
38        points: Ps,
39        surface: Handle<Surface>,
40        core: &mut Core,
41    ) -> Region
42    where
43        P: Into<Point<2>>,
44        Ps: IntoIterator<Item = P>,
45        Ps::IntoIter: Clone + ExactSizeIterator,
46    {
47        let exterior = Cycle::polygon(points, surface, core).insert(core);
48        Region::new(exterior, [])
49    }
50}
