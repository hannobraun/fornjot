11/// Build a [`HalfEdge`]
12///
13/// See [module-level documentation] for context.
14///
15/// [module-level documentation]: super
16pub trait BuildHalfEdge {
17    /// Create a half-edge that is not joined to a sibling
18    fn unjoined(core: &mut Core) -> HalfEdge {
19        let curve = Curve::new().insert(core);
20        let start_vertex = Vertex::new().insert(core);
21
22        HalfEdge::new(curve, start_vertex)
23    }
24
25    /// Create a half-edge from its sibling
26    fn from_sibling(
27        sibling: &Handle<HalfEdge>,
28        start_vertex: Handle<Vertex>,
29        core: &mut Core,
30    ) -> Handle<HalfEdge> {
31        HalfEdge::new(sibling.curve().clone(), start_vertex).insert(core)
32    }
33
34    /// Create an arc
35    ///
36    /// # Panics
37    ///
38    /// Panics if the given angle is not within the range (-2pi, 2pi) radians.
39    fn arc(
40        start: impl Into<Point<2>>,
41        end: impl Into<Point<2>>,
42        angle_rad: impl Into<Scalar>,
43        surface: Handle<Surface>,
44        core: &mut Core,
45    ) -> (Handle<HalfEdge>, CurveBoundary<Point<1>>) {
46        let angle_rad = angle_rad.into();
47        if angle_rad <= -Scalar::TAU || angle_rad >= Scalar::TAU {
48            panic!("arc angle must be in the range (-2pi, 2pi) radians");
49        }
50
51        let arc = Arc::from_endpoints_and_angle(start, end, angle_rad);
52
53        let path = Path::circle_from_center_and_radius(arc.center, arc.radius);
54        let boundary = CurveBoundary {
55            inner: [arc.start_angle, arc.end_angle]
56                .map(|coord| Point::from([coord])),
57        };
58
59        let half_edge = HalfEdge::unjoined(core).insert(core);
60
61        core.layers.geometry.define_curve(
62            half_edge.curve().clone(),
63            surface,
64            LocalCurveGeom { path },
65        );
66
67        (half_edge, boundary)
68    }
69
70    /// Create a line segment
71    fn line_segment(
72        points_surface: [impl Into<Point<2>>; 2],
73        surface: Handle<Surface>,
74        core: &mut Core,
75    ) -> (Handle<HalfEdge>, CurveBoundary<Point<1>>) {
76        let boundary = CurveBoundary::default();
77
78        let half_edge = HalfEdge::unjoined(core).insert(core);
79
80        half_edge.curve().clone().make_line_on_surface(
81            points_surface,
82            boundary,
83            surface.clone(),
84            &mut core.layers.geometry,
85        );
86
87        (half_edge, boundary)
88    }
89}
90
91impl BuildHalfEdge for HalfEdge {}
