17/// # Sweep a [`Path`]
18///
19/// See [module documentation] for more information.
20///
21/// [module documentation]: super
22pub trait SweepSurfacePath {
23    /// # Sweep the surface path
24    ///
25    /// Requires a reference to the surface that the path is defined on.
26    ///
27    ///
28    /// ## Implementation Note
29    ///
30    /// Sweeping a `SurfacePath` that is defined on a curved surface is
31    /// currently not supported:
32    /// <https://github.com/hannobraun/fornjot/issues/1112>
33    fn sweep_surface_path(
34        &self,
35        surface: &SweptCurve,
36        path: impl Into<Vector<3>>,
37        core: &mut Core,
38    ) -> Handle<Surface>;
39}
40
41impl SweepSurfacePath for Path<2> {
42    fn sweep_surface_path(
43        &self,
44        surface: &SweptCurve,
45        path: impl Into<Vector<3>>,
46        core: &mut Core,
47    ) -> Handle<Surface> {
48        let SweptCurve { u, .. } = surface;
49        match u {
50            Path::Circle(_) => {
51                // Sweeping a `Curve` creates a `Surface`. The u-axis of that
52                // `Surface` is a `Path<3>`, which we are computing below. That
53                // computation might or might not work with an arbitrary
54                // surface. Probably not, but I'm not sure.
55                //
56                // What definitely won't work, is computing the bottom edge of
57                // the sweep. The edge sweeping code currently assumes that the
58                // bottom edge is a line (which is true when sweeping from a
59                // flat surface). But if the surface we're sweeping from is
60                // curved, there's simply no way to represent the curve of the
61                // resulting bottom edge.
62                todo!(
63                    "Sweeping a curve that is defined on a curved surface is \
64                    not supported yet."
65                )
66            }
67            Path::Line(_) => {
68                // We're sweeping from a curve on a flat surface, which is
69                // supported. Carry on.
70            }
71        }
72
73        let u = match self {
74            Path::Circle(circle) => {
75                let center = convert_point_surface_to_global(
76                    surface,
77                    circle.center(),
78                    core.tolerance(),
79                    &core.layers.geometry,
80                );
81                let a = convert_vector_surface_to_global(
82                    surface,
83                    circle.a(),
84                    core.tolerance(),
85                    &core.layers.geometry,
86                );
87                let b = convert_vector_surface_to_global(
88                    surface,
89                    circle.b(),
90                    core.tolerance(),
91                    &core.layers.geometry,
92                );
93
94                let circle = Circle::new(center, a, b);
95
96                Path::Circle(circle)
97            }
98            Path::Line(line) => {
99                let origin = convert_point_surface_to_global(
100                    surface,
101                    line.origin(),
102                    core.tolerance(),
103                    &core.layers.geometry,
104                );
105                let direction = convert_vector_surface_to_global(
106                    surface,
107                    line.direction(),
108                    core.tolerance(),
109                    &core.layers.geometry,
110                );
111
112                let line = Line::from_origin_and_direction(origin, direction);
113
114                Path::Line(line)
115            }
116        };
117
118        Surface::from_uv(u, path, core)
119    }
120}
