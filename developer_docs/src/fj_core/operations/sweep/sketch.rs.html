12/// # Sweep a [`Sketch`]
13///
14/// See [module documentation] for more information.
15///
16/// [module documentation]: super
17pub trait SweepSketch {
18    /// # Sweep the [`Sketch`]
19    ///
20    /// Requires `path` to point towards the back of `surface`. If one of them
21    /// is fixed, make sure to adapt the other one accordingly.
22    ///
23    /// Not following this requirement will produce an invalid shape that
24    /// _should_ fail validation.
25    ///
26    /// ## Implementation Note
27    ///
28    /// The above requirement is a bit draconian. It would be much nicer, if
29    /// this operation just worked, regardless of the relation of `path` and
30    /// `surface`, and in fact, a previous version of it did.
31    ///
32    /// However, this previous version also made some undocumented assumption
33    /// that didn't hold in the general case. It was also getting in the way of
34    /// introducing the new geometry system.
35    ///
36    /// The decision was made that, for now, simplifying this operation and
37    /// putting more requirements on the caller, was the right call. Once the
38    /// new geometry system is in place, we'll hopefully be in a position to
39    /// improve the sweep operation substantially.
40    fn sweep_sketch(
41        &self,
42        surface: Handle<Surface>,
43        path: impl Into<Vector<3>>,
44        core: &mut Core,
45    ) -> Solid;
46}
47
48impl SweepSketch for Sketch {
49    fn sweep_sketch(
50        &self,
51        surface: Handle<Surface>,
52        path: impl Into<Vector<3>>,
53        core: &mut Core,
54    ) -> Solid {
55        let path = path.into();
56        let mut cache = SweepCache::default();
57
58        let mut shells = Vec::new();
59        for region in self.regions() {
60            for cycle in region.all_cycles() {
61                for half_edge in cycle.half_edges() {
62                    let curve_geom = core
63                        .layers
64                        .geometry
65                        .of_curve(half_edge.curve())
66                        .unwrap()
67                        .local_on(self.surface())
68                        .unwrap();
69
70                    core.layers.geometry.define_curve(
71                        half_edge.curve().clone(),
72                        surface.clone(),
73                        curve_geom.clone(),
74                    );
75                }
76            }
77
78            let face = Face::new(surface.clone(), region.clone()).insert(core);
79            let shell = face.sweep_face(path, &mut cache, core).insert(core);
80            shells.push(shell);
81        }
82
83        Solid::new(shells)
84    }
85}
