20/// # Sweep a [`HalfEdge`]
21///
22/// See [module documentation] for more information.
23///
24/// [module documentation]: super
25pub trait SweepHalfEdge {
26    /// # Sweep the [`HalfEdge`]
27    ///
28    /// Returns a face, the result of sweeping the edge, as well as the top edge
29    /// of that face, i.e. the edge that is the version of the original edge
30    /// that was translated along the sweep path.
31    ///
32    /// In addition to the usual arguments that many sweep operations require,
33    /// some other ones are needed:
34    ///
35    /// - `end_vertex`, the vertex where the half-edge ends. This is the start
36    ///   vertex of the next half-edge in the cycle.
37    /// - The `surface` that the half-edge is defined on.
38    /// - The `color` of the resulting face, if applicable
39    fn sweep_half_edge(
40        &self,
41        end_vertex: Handle<Vertex>,
42        surface: Handle<Surface>,
43        color: Option<Color>,
44        path: impl Into<Vector<3>>,
45        cache: &mut SweepCache,
46        core: &mut Core,
47    ) -> SweptHalfEdge;
48}
49
50impl SweepHalfEdge for Handle<HalfEdge> {
51    fn sweep_half_edge(
52        &self,
53        end_vertex: Handle<Vertex>,
54        surface: Handle<Surface>,
55        color: Option<Color>,
56        path: impl Into<Vector<3>>,
57        cache: &mut SweepCache,
58        core: &mut Core,
59    ) -> SweptHalfEdge {
60        let path = path.into();
61
62        let boundary = [
63            core.layers
64                .geometry
65                .of_vertex(self.start_vertex())
66                .unwrap()
67                .local_on(self.curve())
68                .unwrap()
69                .position,
70            core.layers
71                .geometry
72                .of_vertex(&end_vertex)
73                .unwrap()
74                .local_on(self.curve())
75                .unwrap()
76                .position,
77        ];
78        let curve_geom = core
79            .layers
80            .geometry
81            .of_curve(self.curve())
82            .unwrap()
83            .local_on(&surface)
84            .unwrap()
85            .clone();
86        let surface_geom = *core.layers.geometry.of_surface(&surface);
87        let surface =
88            curve_geom
89                .path
90                .sweep_surface_path(&surface_geom, path, core);
91
92        // Next, we need to define the boundaries of the face. Let's start with
93        // the global vertices and edges.
94        let (vertices, curves) = {
95            let [a, b] = [self.start_vertex().clone(), end_vertex];
96            let (curve_up, c) = b.clone().sweep_vertex(cache, core);
97            let (curve_down, d) = a.clone().sweep_vertex(cache, core);
98
99            (
100                [a, b, c, d],
101                [
102                    Some(self.curve().clone()),
103                    Some(curve_up),
104                    None,
105                    Some(curve_down),
106                ],
107            )
108        };
109
110        // Let's figure out the surface coordinates of the edge vertices.
111        let surface_points = {
112            let [a, b] = boundary;
113
114            [
115                [a.t, Scalar::ZERO],
116                [b.t, Scalar::ZERO],
117                [b.t, Scalar::ONE],
118                [a.t, Scalar::ONE],
119            ]
120            .map(Point::from)
121        };
122        let surface_points_next = {
123            let mut points = surface_points;
124            points.rotate_left(1);
125            points
126        };
127
128        // Now, the boundaries of each edge.
129        let boundaries = {
130            let [a, b] = boundary;
131            let [c, d] = [0., 1.].map(|coord| Point::from([coord]));
132
133            [[a, b], [c, d], [b, a], [d, c]]
134        };
135
136        let vertices_pairs = {
137            let [a, b, c, d] = vertices;
138            [
139                [a.clone(), b.clone()],
140                [b, c.clone()],
141                [c, d.clone()],
142                [d, a],
143            ]
144        };
145
146        let mut exterior = Cycle::empty();
147
148        // Armed with all of that, we're ready to create the edges.
149        let [_edge_bottom, _edge_up, edge_top, _edge_down] = boundaries
150            .zip_ext(surface_points)
151            .zip_ext(surface_points_next)
152            .zip_ext(vertices_pairs)
153            .zip_ext(curves)
154            .map(
155                |(
156                    (((boundary, start), end), [start_vertex, end_vertex]),
157                    curve,
158                )| {
159                    let boundary = CurveBoundary { inner: boundary };
160
161                    let curve = curve
162                        .unwrap_or_else(|| Curve::new().insert(core))
163                        .make_line_on_surface(
164                            [start, end],
165                            boundary,
166                            surface.clone(),
167                            &mut core.layers.geometry,
168                        );
169
170                    {
171                        let [start, end] = boundary.inner;
172
173                        for (point, vertex) in
174                            [(start, start_vertex.clone()), (end, end_vertex)]
175                        {
176                            core.layers.geometry.define_vertex(
177                                vertex,
178                                curve.clone(),
179                                LocalVertexGeom { position: point },
180                            );
181                        }
182                    }
183
184                    let half_edge = HalfEdge::unjoined(core)
185                        .update_start_vertex(|_, _| start_vertex, core)
186                        .update_curve(|_, _| curve.clone(), core)
187                        .insert(core);
188
189                    exterior =
190                        exterior.add_half_edges([half_edge.clone()], core);
191
192                    half_edge
193                },
194            );
195
196        let exterior = exterior.insert(core);
197        let region = Region::new(exterior, []).insert(core);
198
199        if let Some(color) = color {
200            region.set_color(color, core);
201        }
202
203        let face = Face::new(surface, region);
204        let [_, _, top_boundary, _] = boundaries;
205
206        SweptHalfEdge {
207            face,
208            top_half_edge: edge_top,
209            top_boundary: CurveBoundary {
210                inner: top_boundary,
211            },
212        }
213    }
214}
