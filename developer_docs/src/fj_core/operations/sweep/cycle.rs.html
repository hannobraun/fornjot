17/// # Sweep a [`Cycle`]
18///
19/// See [module documentation] for more information.
20///
21/// [module documentation]: super
22pub trait SweepCycle {
23    /// # Sweep the [`Cycle`]
24    ///
25    /// Sweep the cycle into a set of connected faces. Each half-edge in the
26    /// cycle is swept into a face, meaning all resulting faces form a connected
27    /// set of side walls.
28    ///
29    /// Requires the surface that the half-edges of the cycle are defined in,
30    /// and optionally the color of the created faces.
31    ///
32    /// There is no face at the "top" (the end of the sweep path), as we don't
33    /// have enough information here to create that. We just have a single
34    /// cycle, and don't know whether that is supposed to be the only cycle
35    /// within a potential top face, or whether it's an exterior or interior
36    /// one.
37    ///
38    /// For the same reason, there also is no "bottom" face. Additionally,
39    /// whether a bottom face is even desirable depends on the context this
40    /// operation is called in, and therefore falls outside of its scope.
41    fn sweep_cycle(
42        &self,
43        bottom_surface: Handle<Surface>,
44        top_surface: Handle<Surface>,
45        color: Option<Color>,
46        path: impl Into<Vector<3>>,
47        cache: &mut SweepCache,
48        core: &mut Core,
49    ) -> SweptCycle;
50}
51
52impl SweepCycle for Cycle {
53    fn sweep_cycle(
54        &self,
55        bottom_surface: Handle<Surface>,
56        top_surface: Handle<Surface>,
57        color: Option<Color>,
58        path: impl Into<Vector<3>>,
59        cache: &mut SweepCache,
60        core: &mut Core,
61    ) -> SweptCycle {
62        let path = path.into();
63
64        let mut faces = Vec::new();
65        let mut top_half_edges = Vec::new();
66
67        for bottom_half_edge_pair in self.half_edges().pairs() {
68            let (bottom_half_edge, bottom_half_edge_next) =
69                bottom_half_edge_pair;
70
71            let swept_half_edge = bottom_half_edge.sweep_half_edge(
72                bottom_half_edge_next.start_vertex().clone(),
73                bottom_surface.clone(),
74                color,
75                path,
76                cache,
77                core,
78            );
79
80            faces.push(swept_half_edge.face);
81
82            // The order of these top half-edges is going to be important later,
83            // so let's make sure we understand what's going on:
84            //
85            // - We are iterating through the bottom half-edges here. That means
86            //   the order of those bottom half-edges is natural, as we'd expect
87            //   it:
88            //   - We see them in the order that they appear in the cycle.
89            //   - Each half-edge we see ends where the next one starts.
90            // - By sweeping the bottom half-edges, we are creating a top half-
91            //   edges that have opposite orientation.
92            // - And yet we're adding them to a list, in the same order that we
93            //   iterate over the bottom half-edges.
94            // - As a result, the order of the list is unnatural, going against
95            //   expectations:
96            //   - This is the opposite order than the one in which they'll
97            //     appear within a cycle eventually.
98            //   - Each half-edge ends where the _previous_ one (in the list)
99            //     starts.
100            top_half_edges.push((
101                swept_half_edge.top_half_edge,
102                swept_half_edge.top_boundary,
103                core.layers
104                    .geometry
105                    .of_curve(bottom_half_edge.curve())
106                    .unwrap()
107                    .local_on(&bottom_surface)
108                    .unwrap()
109                    .clone(),
110            ));
111        }
112
113        let top_half_edges = top_half_edges
114            .into_iter()
115            .circular_tuple_windows()
116            .map(
117                |(
118                    (half_edge, boundary, curve_geom),
119                    (next_half_edge, _, _),
120                )| {
121                    let [start, end] = boundary.inner;
122
123                    for (point, vertex) in [
124                        (start, half_edge.start_vertex()),
125                        (end, next_half_edge.start_vertex()),
126                    ] {
127                        core.layers.geometry.define_vertex(
128                            vertex.clone(),
129                            half_edge.curve().clone(),
130                            LocalVertexGeom { position: point },
131                        );
132                    }
133
134                    (half_edge, curve_geom)
135                },
136            )
137            .collect::<Vec<_>>();
138
139        // The half-edges within `top_half_edges` which we're passing into
140        // `add_joined_edges` are in unnatural order, as per the comment above.
141        // This happens to be exactly the order that `add_joined_edges` wants
142        // them to be in, so it works out.
143        let top_cycle = Cycle::empty().add_joined_half_edges(
144            top_half_edges,
145            top_surface,
146            core,
147        );
148
149        SweptCycle { faces, top_cycle }
150    }
151}

153/// The result of sweeping a [`Cycle`]
154///
155/// See [`SweepCycle`].
156pub struct SweptCycle {
157    /// The faces created by sweeping each half-edge of the cycle
158    ///
159    /// See [`SweepCycle::sweep_cycle`] for more information.
160    pub faces: Vec<Face>,
161
162    /// A cycle made up of the "top" half-edges of the resulting faces
163    ///
164    /// "Top" here refers to the place that the sweep path points to from the
165    /// original cycle. Essentially, this is a translated (along the sweep path)
166    /// and reversed version of the original cycle.
167    pub top_cycle: Cycle,
168}
