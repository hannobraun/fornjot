3/// A generic layer, which controls access to layer state
4///
5/// `Layer` is a generic wrapper around some state and controls access to it. It
6/// [`Deref`]s to the state it wraps, for easy read access, but prevents any
7/// direct write access.
8///
9/// Instead, each write access to state is reified as a command, which are
10/// processed by [`Layer::process_command`]. Processing a command can result in
11/// any number of events, which can then be used as commands for other layers.
12///
13/// This design takes inspiration from, and uses the nomenclature of, this
14/// article:
15/// <https://thinkbeforecoding.com/post/2021/12/17/functional-event-sourcing-decider>
16pub struct Layer<S> {
17    state: S,
18}
19
20impl<S> Layer<S> {
21    /// Create an instance of `Layer`
22    pub fn new(state: S) -> Self {
23        Self { state }
24    }
25
26    /// # Process a command without capturing any events
27    ///
28    /// The command is processed synchronously. When this method returns, the
29    /// state has been updated.
30    pub fn process_command<C>(&mut self, command: C) -> C::Result
31    where
32        C: Command<S>,
33    {
34        let mut events = Vec::new();
35        let result = command.decide(&self.state, &mut events);
36
37        for event in events {
38            event.evolve(&mut self.state);
39        }
40
41        result
42    }
43
44    /// # Process a command and capture the events that produces
45    ///
46    /// The command is processed synchronously. When this method returns, the
47    /// state has been updated.
48    pub fn process_command_and_capture_events<C>(
49        &mut self,
50        command: C,
51        events: &mut Vec<C::Event>,
52    ) -> C::Result
53    where
54        C: Command<S>,
55        C::Event: Clone,
56    {
57        let result = command.decide(&self.state, events);
58
59        for event in events.iter().cloned() {
60            event.evolve(&mut self.state);
61        }
62
63        result
64    }
65
66    /// Drop this instance, returning the wrapped state
67    pub fn into_state(self) -> S {
68        self.state
69    }
70}
71
72impl<S> Deref for Layer<S> {
73    type Target = S;
74
75    fn deref(&self) -> &Self::Target {
76        &self.state
77    }
78}
79
80impl<S> Default for Layer<S>
81where
82    S: Default,
83{
84    fn default() -> Self {
85        Self::new(S::default())
86    }
87}

89/// A command that encodes a request to update a layer's state
90pub trait Command<S> {
91    /// The direct result of processing a command that is returned to the caller
92    ///
93    /// Changes to the state that result from a command are encoded as events
94    /// (see [`Command::Event`]). In addition to that, a command may return
95    /// information to the caller, and `Result` defines the type of that
96    /// information.
97    type Result;
98
99    /// An event that encodes a change to the state
100    ///
101    /// Events are produced by [`Command::decide`] and processed by
102    /// [`Event::evolve`].
103    type Event: Event<S>;
104
105    /// Decide which events to produce, given the command and provided state
106    ///
107    /// If the command must result in changes to the state, any number of events
108    /// that describe these state changes can be produced.
109    fn decide(self, state: &S, events: &mut Vec<Self::Event>) -> Self::Result;
110}
111
112/// An event that encodes a change to a layer's state
113pub trait Event<S> {
114    /// Evolve the provided state
115    ///
116    /// This is the only method that [`Layer`] gives mutable access to the
117    /// state, making sure that all changes to the state are captured as events.
118    ///
119    /// Implementations of this method are supposed to be relatively dumb. Any
120    /// decisions that go into updating the state should be made in
121    /// [`Command::decide`], and encoded into the event.
122    fn evolve(self, state: &mut S);
123}
