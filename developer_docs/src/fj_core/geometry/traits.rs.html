1//! # Traits that abstract over curve or surface geometry
2//!
3//! Fornjot's geometry is built on the concept of a uniform representation:
4//! Polylines to represent curves and triangle meshes to represent surfaces. The
5//! traits in this module provide the interface between this uniform
6//! representation and specific geometry code.
7//!
8//! ## Determinism
9//!
10//! Uniform representation must be deterministic. That means a given geometric
11//! curve or surface, at a given tolerance, must generate the same uniform
12//! representation, regardless of where it is queried, and in which order.
13//!
14//! ## Implementation Note
15//!
16//! As of this writing, the transition from the previous, more limited, geometry
17//! system to the new one based on uniform representation is still ongoing. As a
18//! result of that, this module might still be incomplete.
19

27/// # Generate polylines, the uniform representation of curve geometry
28///
29/// This trait provides a generic and uniform interface to curve geometry. It is
30/// implemented by types that represent specific kinds of curve geometry.
31///
32/// It is generic over the dimensionality of the generated polyline. Typically,
33/// two variants should be implemented per curve geometry type:
34///
35/// - `GenPolyline<2>` for surface-local geometry.
36/// - `GenPolyline<3>` for global 3D geometry.
37pub trait GenPolyline<const D: usize> {
38    /// # Access the origin of the curve
39    fn origin(&self) -> Point<D>;
40
41    /// # Compute a line segment to approximate the curve at this point
42    ///
43    /// ## Degenerate Case
44    ///
45    /// If the curve requires no approximation (meaning it is a line), then per
46    /// convention, a degenerate line segment is returned, that collapses to the
47    /// provided point.
48    fn line_segment_at(
49        &self,
50        point_curve: Point<1>,
51        tolerance: Tolerance,
52    ) -> LineSegment<D>;
53
54    /// # Generate a polyline within the provided boundary
55    fn generate_polyline(
56        &self,
57        boundary: CurveBoundary<Point<1>>,
58        tolerance: Tolerance,
59    ) -> Vec<Point<1>>;
60}
61
62// This implementation is temporary, to ease the transition towards a curve
63// geometry trait. Eventually, `CurveGeom2` is expected to replace `Path`.
64impl<const D: usize> GenPolyline<D> for Path<D> {
65    fn origin(&self) -> Point<D> {
66        match self {
67            Self::Circle(circle) => circle.origin(),
68            Self::Line(line) => line.origin(),
69        }
70    }
71
72    fn line_segment_at(
73        &self,
74        point_curve: Point<1>,
75        tolerance: Tolerance,
76    ) -> LineSegment<D> {
77        match self {
78            Self::Circle(circle) => {
79                circle.line_segment_at(point_curve, tolerance)
80            }
81            Self::Line(line) => line.line_segment_at(point_curve, tolerance),
82        }
83    }
84
85    fn generate_polyline(
86        &self,
87        boundary: CurveBoundary<Point<1>>,
88        tolerance: Tolerance,
89    ) -> Vec<Point<1>> {
90        match self {
91            Self::Circle(circle) => {
92                circle.generate_polyline(boundary, tolerance)
93            }
94            Self::Line(line) => line.generate_polyline(boundary, tolerance),
95        }
96    }
97}

99/// # Generate triangle meshes, the uniform representation of surface geometry
100pub trait GenTriMesh {
101    /// # Access the origin of the surface
102    fn origin(&self, geometry: &Geometry) -> Point<3>;
103
104    /// # Return the triangle at the provided point on the surface
105    ///
106    /// Select a triangle of the surface's triangle mesh representation, the one
107    /// at the provided surface point. Return that triangle, as well as the
108    /// barycentric coordinates of the provided point on the triangle.
109    ///
110    /// ## Triangle Size and Validity
111    ///
112    /// If a surface is curved along both axes, the triangle's size is chosen
113    /// such, that it approximates the surface, with the maximum allowed
114    /// deviation of the actual surface defined by the provided tolerance
115    /// argument.
116    ///
117    /// Otherwise, the size of the returned triangle is at least partially
118    /// arbitrary. Take the extreme case of a plane: Since it is not curved at
119    /// all, the returned triangle can be arbitrarily large.
120    ///
121    /// However, since surfaces are infinite, and we can't represent infinite
122    /// triangles, there is no sensible upper bound for the size. Instead, to
123    /// prevent an arbitrary choice for the size of triangles, which would imply
124    /// properties of the surface that are not true, and might therefore be
125    /// confusing, the triangles returned by this function have a length of zero
126    /// along axes that do not require approximation.
127    ///
128    /// The most extreme case would be a plane, for which the returned triangle
129    /// is collapsed to a point. For a cylinder, the triangle would have the
130    /// appropriate width to approximate the curved axis given the provided
131    /// tolerance, while having zero height.
132    ///
133    /// ## Implementation Note
134    ///
135    /// At the time this was written, there was no dedicated type to represent
136    /// barycentric coordinates. Nor any other code that used them, I think.
137    ///
138    /// If this changes, and a special type for barycentric coordinates is
139    /// added, it would make sense to return that here.
140    fn triangle_at(
141        &self,
142        point_surface: Point<2>,
143        tolerance: Tolerance,
144        geometry: &Geometry,
145    ) -> (Triangle<3>, [Scalar; 3]);
146
147    /// # Generated a triangle mesh within the provided boundary
148    fn generate_tri_mesh(
149        &self,
150        boundary: Aabb<2>,
151        tolerance: Tolerance,
152        geometry: &Geometry,
153    ) -> Vec<Point<2>>;
154}
155
156impl<T> GenTriMesh for T
157where
158    T: Deref,
159    T::Target: GenTriMesh,
160{
161    fn origin(&self, geometry: &Geometry) -> Point<3> {
162        self.deref().origin(geometry)
163    }
164
165    fn triangle_at(
166        &self,
167        point_surface: Point<2>,
168        tolerance: Tolerance,
169        geometry: &Geometry,
170    ) -> (Triangle<3>, [Scalar; 3]) {
171        self.deref().triangle_at(point_surface, tolerance, geometry)
172    }
173
174    fn generate_tri_mesh(
175        &self,
176        boundary: Aabb<2>,
177        tolerance: Tolerance,
178        geometry: &Geometry,
179    ) -> Vec<Point<2>> {
180        self.deref()
181            .generate_tri_mesh(boundary, tolerance, geometry)
182    }
183}
