1use fj_interop::Tolerance;
2use fj_math::{Aabb, Point, Scalar, Triangle, Vector};
3
4use crate::geometry::{
5    Geometry, Path,
6    traits::{GenPolyline, GenTriMesh},
7};
8
9/// # A surface that is a curve, swept along a path
10#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash, Ord, PartialOrd)]
11pub struct SweptCurve {
12    /// The u-axis of the surface
13    pub u: Path<3>,
14
15    /// The v-axis of the surface
16    pub v: Vector<3>,
17}
18
19impl GenTriMesh for SweptCurve {
20    fn origin(&self, _: &Geometry) -> Point<3> {
21        self.u.origin()
22    }
23
24    fn triangle_at(
25        &self,
26        point_surface: Point<2>,
27        tolerance: Tolerance,
28        _: &Geometry,
29    ) -> (Triangle<3>, [Scalar; 3]) {
30        let [a, b] = self
31            .u
32            .line_segment_at(Point::from([point_surface.u]), tolerance)
33            .points
34            .map(|point_global| point_global + self.v * point_surface.v);
35
36        let c = a + (b - a) / 2.;
37        let triangle = Triangle::from([a, b, c]);
38
39        let barycentric_coords = [1. / 3.; 3].map(Into::into);
40        (triangle, barycentric_coords)
41    }
42
43    fn generate_tri_mesh(
44        &self,
45        boundary: Aabb<2>,
46        tolerance: Tolerance,
47        _: &Geometry,
48    ) -> Vec<Point<2>> {
49        let boundary_curve = [[boundary.min.u], [boundary.max.u]];
50        let points_curve =
51            self.u.generate_polyline(boundary_curve.into(), tolerance);
52
53        points_curve
54            .iter()
55            .copied()
56            .map(|point| [point.t, Scalar::ZERO])
57            .chain(
58                points_curve
59                    .iter()
60                    .copied()
61                    .map(|point| [point.t, self.v.magnitude()]),
62            )
63            .map(Point::from)
64            .collect()
65    }
66}
