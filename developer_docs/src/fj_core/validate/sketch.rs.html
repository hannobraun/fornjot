15impl Validate for Sketch {
16    fn validate(
17        &self,
18        config: &ValidationConfig,
19        errors: &mut Vec<ValidationError>,
20        geometry: &Geometry,
21    ) {
22        errors.extend(
23            AdjacentHalfEdgesNotConnected::check(self, geometry, config)
24                .map(Into::into),
25        );
26        errors.extend(
27            MultipleReferencesToObject::<Cycle, Region>::check(
28                self, geometry, config,
29            )
30            .map(Into::into),
31        );
32        errors.extend(
33            MultipleReferencesToObject::<HalfEdge, Cycle>::check(
34                self, geometry, config,
35            )
36            .map(Into::into),
37        );
38        SketchValidationError::check_exterior_cycles(
39            self, geometry, config, errors,
40        );
41        SketchValidationError::check_interior_cycles(
42            self, geometry, config, errors,
43        );
44    }
45}

47/// [`Sketch`] validation failed
48#[derive(Clone, Debug, thiserror::Error)]
49pub enum SketchValidationError {
50    /// Region within sketch has exterior cycle with clockwise winding
51    #[error(
52        "Exterior cycle within sketch region has clockwise winding\n
53        Cycle: {cycle:#?}"
54    )]
55    ClockwiseExteriorCycle {
56        /// Cycle with clockwise winding
57        cycle: Handle<Cycle>,
58    },
59
60    /// Region within sketch has interior cycle with counter-clockwise winding
61    #[error(
62        "Interior cycle within sketch region has counter-clockwise winding\n
63        Cycle: {cycle:#?}"
64    )]
65    CounterClockwiseInteriorCycle {
66        /// Cycle with counter-clockwise winding
67        cycle: Handle<Cycle>,
68    },
69}
70
71impl SketchValidationError {
72    fn check_exterior_cycles(
73        sketch: &Sketch,
74        geometry: &Geometry,
75        _config: &ValidationConfig,
76        errors: &mut Vec<ValidationError>,
77    ) {
78        sketch.regions().iter().for_each(|region| {
79            let cycle = region.exterior();
80            if cycle.winding(geometry, sketch.surface()) == Winding::Cw {
81                errors.push(ValidationError::Sketch(
82                    SketchValidationError::ClockwiseExteriorCycle {
83                        cycle: cycle.clone(),
84                    },
85                ))
86            }
87        });
88    }
89
90    fn check_interior_cycles(
91        sketch: &Sketch,
92        geometry: &Geometry,
93        _config: &ValidationConfig,
94        errors: &mut Vec<ValidationError>,
95    ) {
96        sketch.regions().iter().for_each(|region| {
97            region
98                .interiors()
99                .iter()
100                .filter(|interior| {
101                    interior.winding(geometry, sketch.surface()) == Winding::Ccw
102                })
103                .for_each(|cycle| {
104                    errors.push(ValidationError::Sketch(
105                        SketchValidationError::CounterClockwiseInteriorCycle {
106                            cycle: cycle.clone(),
107                        },
108                    ));
109                })
110        });
111    }
112}

130        let mut core = Core::new();
131
132        let valid = Sketch::circle([0., 0.], 1., &mut core);
133        valid.validate_and_return_first_error(&core.layers.geometry)?;
134
135        let invalid_sketch = valid.update_region(
