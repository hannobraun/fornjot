8/// Queries related to the sibling of a [`HalfEdge`]
9pub trait SiblingOfHalfEdge {
10    /// Indicate whether the provided half-edges are siblings
11    fn are_siblings(&self, a: &Handle<HalfEdge>, b: &Handle<HalfEdge>) -> bool;
12
13    /// Retrieve the sibling of this half-edge
14    ///
15    /// Returns `None`, if the provided half-edge is not part of the object this
16    /// method is called on, or if the provided half-edge has no sibling within
17    /// the object.
18    fn get_sibling_of(&self, half_edge: &Handle<HalfEdge>) -> Option<Sibling>;
19}
20
21impl SiblingOfHalfEdge for Shell {
22    fn are_siblings(&self, a: &Handle<HalfEdge>, b: &Handle<HalfEdge>) -> bool {
23        let same_curve = a.curve().id() == b.curve().id();
24        let same_vertices = {
25            let Some(a_vertices) = self.bounding_vertices_of_half_edge(a)
26            else {
27                return false;
28            };
29            let Some(b_vertices) = self.bounding_vertices_of_half_edge(b)
30            else {
31                return false;
32            };
33
34            a_vertices == b_vertices.reverse()
35        };
36
37        same_curve && same_vertices
38    }
39
40    fn get_sibling_of(&self, half_edge: &Handle<HalfEdge>) -> Option<Sibling> {
41        for face in self.faces() {
42            for cycle in face.region().all_cycles() {
43                for h in cycle.half_edges() {
44                    if self.are_siblings(half_edge, h) {
45                        return Some(Sibling {
46                            sibling: h.clone(),
47                            cycle: cycle.clone(),
48                            face: face.clone(),
49                        });
50                    }
51                }
52            }
53        }
54
55        None
56    }
57}

59/// The sibling of a half-edge, plus some extra information

64    /// The cycle in which the sibling was found
65    pub cycle: Handle<Cycle>,
66
67    /// The face in which the sibling was found
68    pub face: Handle<Face>,
69}
