1use crate::{
2    storage::Handle,
3    topology::{Cycle, HalfEdge, Shell},
4};
5
6/// Query to find the cycle that a half-edge is part of
7pub trait CycleOfHalfEdge {
8    /// Find the cycle that a half-edge is part of
9    fn find_cycle_of_half_edge(
10        &self,
11        half_edge: &Handle<HalfEdge>,
12    ) -> Option<Handle<Cycle>>;
13}
14
15impl CycleOfHalfEdge for Shell {
16    fn find_cycle_of_half_edge(
17        &self,
18        half_edge: &Handle<HalfEdge>,
19    ) -> Option<Handle<Cycle>> {
20        for face in self.faces() {
21            for cycle in face.region().all_cycles() {
22                for h in cycle.half_edges() {
23                    if h == half_edge {
24                        return Some(cycle.clone());
25                    }
26                }
27            }
28        }
29
30        None
31    }
32}
