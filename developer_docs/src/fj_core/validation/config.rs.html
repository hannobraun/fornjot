1use fj_interop::Tolerance;
2use fj_math::Scalar;
3
4/// Configuration required for the validation process
5#[derive(Debug, Clone, Copy)]
6pub struct ValidationConfig {
7    /// Panic on first validation error, instead of storing it
8    ///
9    /// Validation errors are usually stored in the validation layer, and only
10    /// cause a panic if the validation layer is dropped with unhandled errors.
11    ///
12    /// This provides flexibility in handling validation errors, but can also be
13    /// helpful in understanding them, as experience has shown that the first
14    /// validation error often does not provide a full picture of what's wrong.
15    ///
16    /// However, it can be helpful to get an immediate panic on a validation
17    /// error, to get the code that caused it into a stack trace. This is what
18    /// happens, if this option is set to `true`.
19    ///
20    /// Defaults to `false`.
21    pub panic_on_error: bool,
22
23    /// The tolerance value used for intermediate geometry representation
24    pub tolerance: Tolerance,
25
26    /// The maximum distance between identical objects
27    ///
28    /// Objects that are considered identical might still have a distance
29    /// between them, due to inaccuracies of the numerical representation. If
30    /// that distance is less than the one defined in this field, can not be
31    /// considered identical.
32    pub identical_max_distance: Scalar,
33
34    /// The minimum distance between distinct objects
35    ///
36    /// Objects whose distance is less than the value defined in this field, are
37    /// considered identical.
38    pub distinct_min_distance: Scalar,
39}
40
41impl ValidationConfig {
42    /// Compute validation config from a tolerance value
43    pub fn from_tolerance(tolerance: impl Into<Tolerance>) -> Self {
44        let tolerance = tolerance.into();
45
46        // This value can't be smaller than the tolerance. If it is, we'll get
47        // validation errors everywhere, just from numerical noise.
48        let identical_max_distance = tolerance.inner() * 10.;
49
50        // This value can't be smaller than `identical_max_distance`. Otherwise
51        // we can have distinct points that satisfy this constraint, but must be
52        // considered identical according to the other.
53        //
54        // This factor was chosen pretty arbitrarily and might need to be tuned.
55        let distinct_min_distance = identical_max_distance * 2.;
56
57        Self {
58            panic_on_error: false,
59            tolerance,
60            identical_max_distance,
61            distinct_min_distance,
62        }
63    }
64}
65
66impl Default for ValidationConfig {
67    fn default() -> Self {
68        Self::from_tolerance(0.001)
69    }
70}
