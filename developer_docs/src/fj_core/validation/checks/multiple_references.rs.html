10/// Object that should be exclusively owned by another, is not
11///
12/// Some objects are expected to be "owned" by a single other object. This means
13/// that only one reference to these objects must exist within the topological
14/// object graph.
15#[derive(Clone, Debug, thiserror::Error)]
16pub struct MultipleReferencesToObject<T, U> {
17    object: Handle<T>,
18    referenced_by: Vec<Handle<U>>,
19}
20
21impl<T, U> fmt::Display for MultipleReferencesToObject<T, U>
22where
23    T: fmt::Debug,
24    U: fmt::Debug,
25{
26    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
27        write!(
28            f,
29            "`{}` ({:?}) referenced by multiple `{}` objects ({:?})",
30            type_name_of_val(&self.object),
31            self.object,
32            type_name_of_val(&self.referenced_by),
33            self.referenced_by
34        )
35    }
36}

38impl ValidationCheck<Sketch> for MultipleReferencesToObject<Cycle, Region> {
39    fn check<'r>(
40        object: &'r Sketch,
41        _: &'r Geometry,
42        _: &'r ValidationConfig,
43    ) -> impl Iterator<Item = Self> + 'r {
44        let mut cycles = ReferenceCounter::new();
45
46        for region in object.regions() {
47            for cycle in region.all_cycles() {
48                cycles.count(cycle.clone(), region.clone());
49            }
50        }
51
52        cycles.multiples()
53    }
54}

154#[derive(Default)]
155struct ReferenceCounter<T, U>(HashMap<Handle<T>, Vec<Handle<U>>>);
156
157impl<T, U> ReferenceCounter<T, U> {
158    pub fn new() -> Self {
159        Self(HashMap::new())
160    }
161
162    pub fn count(&mut self, to: Handle<T>, from: Handle<U>) {
163        self.0.entry(to).or_default().push(from);
164    }
165
166    pub fn multiples(
167        self,
168    ) -> impl Iterator<Item = MultipleReferencesToObject<T, U>> {
169        self.0
170            .into_iter()
171            .filter(|(_, referenced_by)| referenced_by.len() > 1)
172            .map(|(object, referenced_by)| MultipleReferencesToObject {
173                object: object.clone(),
174                referenced_by: referenced_by.to_vec(),
175            })
176    }
177}
