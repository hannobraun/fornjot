10/// Interior [`Cycle`] of [`Face`] has invalid winding
11///
12/// The winding of a face's exterior cycle is part of what defines the
13/// orientation of that face. The winding of the interior cycle has no such
14/// meaning attached to it, but it can't be arbitrary either. Triangulation, for
15/// example, might need to assume that it is the opposite of the exterior
16/// winding.
17///
18/// This validation check ensures just that: that the winding of the interior
19/// cycles of a face is the opposite of the winding of that face's exterior
20/// cycle.
21#[derive(Clone, Debug, thiserror::Error)]
22#[error(
23    "Interior of `Face` has invalid winding; must be opposite of exterior\n\
24    - Winding of exterior cycle: {exterior_winding:#?}\n\
25    - Interior cycle with invalid winding: {interior_cycle:#?}\n\
26    - Winding of invalid interior cycle: {interior_winding:#?}"
27)]
28pub struct InteriorCycleHasInvalidWinding {
29    /// The winding of the [`Face`]'s exterior cycle
30    pub exterior_winding: Winding,
31
32    /// The interior cycle with invalid winding
33    pub interior_cycle: Handle<Cycle>,
34
35    /// The winding of the invalid interior cycle
36    pub interior_winding: Winding,
37}
38
39impl ValidationCheck<Face> for InteriorCycleHasInvalidWinding {
40    fn check<'r>(
41        object: &'r Face,
42        geometry: &'r Geometry,
43        _: &'r ValidationConfig,
44    ) -> impl Iterator<Item = Self> + 'r {
45        object.region().interiors().iter().filter_map(|interior| {
46            let exterior = object.region().exterior();
47
48            if exterior.half_edges().is_empty()
49                || interior.half_edges().is_empty()
50            {
51                // Can't determine winding, if the cycle has no edges. Sounds
52                // like a job for a different validation check.
53                return None;
54            }
55
56            let exterior_winding = exterior.winding(geometry, object.surface());
57            let interior_winding = interior.winding(geometry, object.surface());
58
59            if exterior_winding == interior_winding {
60                return Some(InteriorCycleHasInvalidWinding {
61                    exterior_winding,
62                    interior_cycle: interior.clone(),
63                    interior_winding,
64                });
65            }
66
67            None
68        })
69    }
70}

87    #[test]
88    fn interior_winding() -> anyhow::Result<()> {
89        let mut core = Core::new();
90
91        let surface = core.layers.topology.surfaces.xy_plane();
92        let valid = Face::polygon(
93            surface.clone(),
94            [[0., 0.], [3., 0.], [0., 3.]],
95            &mut core,
96        )
97        .update_region(
98            |region, core| {
99                region.add_interiors(
100                    [Cycle::polygon(
101                        [[1., 1.], [1., 2.], [2., 1.]],
102                        surface,
103                        core,
104                    )],
105                    core,
106                )
107            },
108            &mut core,
109        );
110        InteriorCycleHasInvalidWinding::check_and_return_first_error(
111            &valid,
112            &core.layers.geometry,
113        )?;
114
115        let invalid = {
116            let interiors = valid
117                .region()
118                .interiors()
119                .iter()
120                .cloned()
121                .map(|cycle| {
122                    cycle
123                        .reverse(&mut core)
124                        .insert(&mut core)
125                        .derive_from(&cycle, &mut core)
126                })
127                .collect::<Vec<_>>();
128
129            let region =
130                Region::new(valid.region().exterior().clone(), interiors)
131                    .insert(&mut core);
132
133            Face::new(valid.surface().clone(), region)
134        };
135        InteriorCycleHasInvalidWinding::check_and_expect_one_error(
136            &invalid,
137            &core.layers.geometry,
138        );
139
140        Ok(())
141    }
