11/// A [`Shell`] contains a [`HalfEdge`] without a sibling
12///
13/// Half-edges that are coincident must reference the same curve. This makes
14/// those half-edges siblings.
15///
16/// In a shell, every half-edge must have a sibling. If that is not the case,
17/// this is a sign of either of the following:
18/// - That the shell is not closed, meaning it has some kind of hole.
19/// - If the shell is closed, that its topological object graph is not valid.
20#[derive(Clone, Debug, thiserror::Error)]
21#[error("Half-edge has no sibling: {half_edge:#?}")]
22pub struct HalfEdgeHasNoSibling {
23    /// The half-edge that does not have a sibling
24    pub half_edge: Handle<HalfEdge>,
25}
26
27impl ValidationCheck<Shell> for HalfEdgeHasNoSibling {
28    fn check<'r>(
29        object: &'r Shell,
30        _: &'r Geometry,
31        _: &'r ValidationConfig,
32    ) -> impl Iterator<Item = Self> + 'r {
33        let mut unmatched_half_edges = BTreeMap::new();
34
35        for face in object.faces() {
36            for cycle in face.region().all_cycles() {
37                for half_edge in cycle.half_edges() {
38                    let curve = half_edge.curve().clone();
39                    let vertices =
40                        cycle.bounding_vertices_of_half_edge(half_edge).expect(
41                            "`half_edge` came from `cycle`, must exist there",
42                        );
43
44                    let key = (curve.clone(), vertices.clone());
45                    let key_reversed = (curve, vertices.reverse());
46
47                    match unmatched_half_edges.remove(&key_reversed) {
48                        Some(sibling) => {
49                            // This must be the sibling of the half-edge we're
50                            // currently looking at. Let's make sure the logic
51                            // we use here to determine that matches the
52                            // "official" definition.
53                            assert!(object.are_siblings(half_edge, sibling));
54                        }
55                        None => {
56                            // If this half-edge has a sibling, we haven't seen
57                            // it yet. Let's store this half-edge then, in case
58                            // we come across the sibling later.
59                            unmatched_half_edges.insert(key, half_edge);
60                        }
61                    }
62                }
63            }
64        }
65
66        unmatched_half_edges
67            .into_values()
68            .cloned()
69            .map(|half_edge| HalfEdgeHasNoSibling { half_edge })
70    }
71}
