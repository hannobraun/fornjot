44    pub fn contains_triangle(&self, triangle: impl Into<Triangle<2>>) -> bool {
45        let [a, b, c] = triangle.into().points;
46
47        let mut might_be_hole = true;
48
49        for &edge in [a, b, c, a].as_slice().array_windows_ext() {
50            let edge = LineSegment::from(edge);
51
52            let is_exterior_edge = self.contains_exterior_edge(edge);
53            let is_interior_edge = self.contains_interior_edge(edge);
54
55            // If the triangle edge is not an interior edge of the polygon, we
56            // can rule out that the triangle is identical with a hole in the
57            // polygon.
58            if !is_interior_edge {
59                might_be_hole = false;
60            }
61
62            // If the triangle edge is an edge of the face, we don't need to
63            // take a closer look.
64            if is_exterior_edge || is_interior_edge {
65                continue;
66            }
67
68            // To determine if the edge is within the polygon, we determine if
69            // its center point is in the polygon.
70            //
71            // Since we already checked above, whether the triangle edge is a
72            // polygon edge (and if we reached this point, it isn't), we don't
73            // need to care about the distinction between "inside the polygon"
74            // and "on the polygon boundary".
75            if !self.contains_point(edge.center()) {
76                // The segment is outside of the face. This means we can throw
77                // away the whole triangle.
78                return false;
79            }
80        }
81
82        // We haven't ruled out that the triangle is a polygon hole. Since we
83        // checked all its edges, this means we now know for certain that is is.
84        if might_be_hole {
85            return false;
86        }
87
88        // If we didn't throw away the triangle up till now, this means all its
89        // edges are within the face.
90        true
91    }
92
93    fn contains_exterior_edge(&self, edge: LineSegment<2>) -> bool {
94        self.exterior.segments().contains(&edge)
95            || self.exterior.segments().contains(&edge.reverse())
96    }
97
98    fn contains_interior_edge(&self, edge: LineSegment<2>) -> bool {
99        let mut contains = false;
100
101        for chain in &self.interiors {
102            contains |= chain.segments().contains(&edge);
103            contains |= chain.segments().contains(&edge.reverse());
104        }
105
106        contains
107    }

109    /// Check whether the polygon contains a point
110    ///
111    /// # Implementation Note
112    ///
113    /// This code is being duplicated by the `Contains<Point<2>>` implementation
114    /// for `Face`. It would be nice to be able to consolidate the duplication,
115    /// but this has turned out to be difficult.
116    fn contains_point(&self, point: impl Into<Point<2>>) -> bool {
117        let ray = HorizontalRayToTheRight {
118            origin: point.into(),
119        };
120
121        let mut num_hits = 0;
122
123        for chain in Some(&self.exterior).into_iter().chain(&self.interiors) {
124            let edges = chain.segments();
125
126            // We need to properly detect the ray passing the boundary at the
127            // "seam" of the polygon, i.e. the vertex between the last and the
128            // first segment. The logic in the loop properly takes care of that,
129            // as long as we initialize the `previous_hit` variable with the
130            // result of the last segment.
131            let mut previous_hit = edges
132                .last()
133                .copied()
134                .and_then(|edge| (&ray, &edge).intersect());
135
136            for edge in edges {
137                let hit = (&ray, &edge).intersect();
138
139                let count_hit = match (hit, previous_hit) {
140                    (
141                        Some(
142                            RaySegmentIntersection::RayStartsOnSegment
143                            | RaySegmentIntersection::RayStartsOnOnFirstVertex
144                            | RaySegmentIntersection::RayStartsOnSecondVertex,
145                        ),
146                        _,
147                    ) => {
148                        // If the ray starts on the boundary of the polygon,
149                        // there's nothing else to check. By the definition of
150                        // this intersection test, the polygon contains the
151                        // point.
152                        return true;
153                    }
154                    (Some(RaySegmentIntersection::RayHitsSegment), _) => {
155                        // We're hitting a segment right-on. Clear case.
156                        true
157                    }
158                    (
159                        Some(RaySegmentIntersection::RayHitsUpperVertex),
160                        Some(RaySegmentIntersection::RayHitsLowerVertex),
161                    )
162                    | (
163                        Some(RaySegmentIntersection::RayHitsLowerVertex),
164                        Some(RaySegmentIntersection::RayHitsUpperVertex),
165                    ) => {
166                        // If we're hitting a vertex, only count it if we've hit
167                        // the other kind of vertex right before.
168                        //
169                        // That means, we're passing through the polygon
170                        // boundary at where two edges touch. Depending on the
171                        // order in which edges are checked, we're seeing this
172                        // as a hit to one edge's lower/upper vertex, then the
173                        // other edge's opposite vertex.
174                        //
175                        // If we're seeing two of the same vertices in a row,
176                        // we're not actually passing through the polygon
177                        // boundary. Then we're just touching a vertex without
178                        // passing through anything.
179                        true
180                    }
181                    (Some(RaySegmentIntersection::RayHitsSegmentAndAreParallel), _) => {
182                        // A parallel edge must be completely ignored. Its
183                        // presence won't change anything, so we can treat it as
184                        // if it wasn't there, and its neighbors were connected
185                        // to each other.
186                        continue;
187                    }
188                    _ => {
189                        // Any other case is not a valid hit.
190                        false
191                    }
192                };
193
194                if count_hit {
195                    num_hits += 1;
196                }
197
198                previous_hit = hit;
199            }
200        }
201
202        num_hits % 2 == 1
203    }
204}
