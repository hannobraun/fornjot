20/// Approximate the provided cycle
21pub fn approx_cycle(
22    cycle: &Cycle,
23    surface: &Handle<Surface>,
24    tolerance: impl Into<Tolerance>,
25    cache: &mut ApproxCache,
26    geometry: &Geometry,
27) -> CycleApprox {
28    let tolerance = tolerance.into();
29
30    let half_edges = cycle
31        .half_edges()
32        .pairs()
33        .map(|(half_edge, next_half_edge)| {
34            let boundary = CurveBoundary {
35                inner: [
36                    geometry
37                        .of_vertex(half_edge.start_vertex())
38                        .unwrap()
39                        .local_on(half_edge.curve())
40                        .unwrap()
41                        .position,
42                    geometry
43                        .of_vertex(next_half_edge.start_vertex())
44                        .unwrap()
45                        .local_on(half_edge.curve())
46                        .unwrap()
47                        .position,
48                ],
49            };
50            let [start_position_curve, _] = boundary.inner;
51
52            let start = approx_vertex(
53                half_edge.start_vertex().clone(),
54                half_edge.curve(),
55                surface,
56                start_position_curve,
57                tolerance,
58                &mut cache.vertex,
59                geometry,
60            );
61
62            approx_half_edge(
63                half_edge,
64                surface,
65                start,
66                boundary,
67                tolerance,
68                &mut cache.curve,
69                geometry,
70            )
71        })
72        .collect();
73
74    CycleApprox { half_edges }
75}
76
77/// An approximation of a [`Cycle`]
78#[derive(Debug, Eq, PartialEq, Hash, Ord, PartialOrd)]
79pub struct CycleApprox {
80    /// The approximated half-edges that make up the approximated cycle
81    pub half_edges: Vec<HalfEdgeApprox>,
82}
83
84impl CycleApprox {
85    /// Compute the points that approximate the cycle
86    pub fn points(&self) -> Vec<ApproxPoint<2>> {
87        let mut points = Vec::new();
88
89        for approx in &self.half_edges {
90            points.extend(approx.points.iter().copied());
91        }
92
93        if let Some(point) = points.first() {
94            points.push(*point);
95        }
96
97        points
98    }
99
100    /// Construct the segments that approximate the cycle
101    pub fn segments(&self) -> Vec<LineSegment<3>> {
102        let mut segments = Vec::new();
103
104        for segment in self.points().windows(2) {
105            // This can't panic, as we passed `2` to `windows`. Can be cleaned
106            // up, once `array_windows` is stable.
107            let segment = [&segment[0], &segment[1]];
108
109            segments.push(LineSegment::from(
110                segment.map(|point| point.global_form),
111            ));
112        }
113
114        segments
115    }
116}
