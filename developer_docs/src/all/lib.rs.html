11pub fn model(core: &mut fj::core::Core) -> Solid {
12    // Just combine all the other models using offsets/rotations that won't
13    // result in neat vertex positions or axis-aligned edges/faces. This is
14    // useful for testing.
15
16    let offset = Vector::from([5., 5., 5.]);
17    let axis = Vector::from([1., 1., 1.]).normalize();
18    let angle_rad = Scalar::PI / 6.;
19
20    let models = [
21        color::model(core),
22        cuboid::model([1., 2., 3.], core),
23        holes::model(0.5, core),
24        spacer::model(2., 1., 1., core),
25        split::model(1., 0.2, core),
26        star::model(5, 2., 1., 1., core),
27        vertices_indices::model(core),
28    ];
29
30    let mut all = Solid::empty();
31
32    for (i, model) in models.into_iter().enumerate() {
33        let f = i as f64;
34
35        let model = model
36            .translate(offset * f, core)
37            .rotate(axis * angle_rad * f, core);
38
39        all = all.merge(&model, core);
40    }
41
42    all
43}
