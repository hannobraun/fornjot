36    /// Export or display a model, according to CLI arguments
37    ///
38    /// This function is intended to be called by applications that define a
39    /// model and want to provide a standardized CLI interface for dealing with
40    /// that model.
41    ///
42    /// This function is used by Fornjot's own testing infrastructure, but is
43    /// useful beyond that, when using Fornjot directly to define a model.
44    pub fn process_model<M>(&mut self, model: &M) -> Result
45    where
46        for<'r> (&'r M, Tolerance): Triangulate,
47        for<'r> &'r M: BoundingVolume<3>,
48    {
49        let args = Args::parse();
50        self.process_model_args(model, args)
51    }
52
53    /// Process a model with pre-parsed arguments
54    ///
55    /// This function is similar to [`Self::process_model`], but accepts pre-parsed arguments
56    /// instead of parsing them from the command line. This is useful when you want to
57    /// extend the standard arguments with your own parameters.
58    pub fn process_model_args<M>(&mut self, model: &M, args: Args) -> Result
59    where
60        for<'r> (&'r M, Tolerance): Triangulate,
61        for<'r> &'r M: BoundingVolume<3>,
62    {
63        tracing_subscriber::registry()
64            .with(tracing_subscriber::fmt::layer())
65            .with(tracing_subscriber::EnvFilter::from_default_env())
66            .init();
67
68        if !args.ignore_validation {
69            self.core.layers.validation.take_errors()?;
70        }
71
72        let aabb = model.aabb(&self.core.layers.geometry).unwrap_or(Aabb {
73            min: Point::origin(),
74            max: Point::origin(),
75        });
76
77        let tolerance = match args.tolerance {
78            None => {
79                // Compute a reasonable default for the tolerance value. To do
80                // this, we just look at the smallest non-zero extent of the
81                // bounding box and divide that by some value.
82                let mut min_extent = Scalar::MAX;
83                for extent in aabb.size().components {
84                    if extent > Scalar::ZERO && extent < min_extent {
85                        min_extent = extent;
86                    }
87                }
88
89                let tolerance = min_extent / Scalar::from_f64(1000.);
90                Tolerance::from_scalar(tolerance)?
91            }
92            Some(user_defined_tolerance) => user_defined_tolerance,
93        };
94
95        let tri_mesh = (model, tolerance).triangulate(&mut self.core);
96
97        if let Some(path) = args.export {
98            export::export(tri_mesh.all_triangles(), &path)?;
99            return Ok(());
100        }
101
102        make_viewer_and_spawn_thread(|viewer| {
103            viewer.open_window().display_mesh(tri_mesh);
104        })?;
105
106        Ok(())
107    }
108}

110/// Return value of [`Instance::process_model`]
111pub type Result = std::result::Result<(), Error>;

113/// Error returned by [`Instance::process_model`]
114#[derive(thiserror::Error)]
115pub enum Error {
116    /// Failed to set up logger
117    #[error("Failed to set up logger")]
118    Tracing(#[from] tracing::subscriber::SetGlobalDefaultError),
119
120    /// Error displaying model
121    #[error("Error displaying model")]
122    Display(#[from] viewer::Error),
123
124    /// Error exporting model
125    #[error("Error exporting model")]
126    Export(#[from] export::Error),
127
128    /// Invalid tolerance
129    #[error(transparent)]
130    Tolerance(#[from] InvalidTolerance),
131
132    /// Unhandled validation errors
133    #[error(transparent)]
134    Validation(#[from] ValidationErrors),
135}

137impl fmt::Debug for Error {
138    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
139        // When returning an error from Rust's `main` function, the runtime uses
140        // the error's `Debug` implementation to display it, not the `Display`
141        // one. This is unfortunate, and forces us to override `Debug` here.
142
143        // We should be able to replace this with `Report`, once it is stable:
144        // https://doc.rust-lang.org/std/error/struct.Report.html
145
146        write!(f, "{self}")?;
147
148        let mut source = self.source();
149
150        if source.is_some() {
151            write!(f, "\n\nCaused by:")?;
152        }
153
154        let mut i = 0;
155        while let Some(s) = source {
156            write!(f, "\n    {i}: {s}")?;
157            source = s.source();
158            i += 1;
159        }
160
161        Ok(())
162    }
