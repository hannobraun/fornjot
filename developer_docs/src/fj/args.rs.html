6/// Standardized CLI for Fornjot models
7///
8/// This is completely optional, as models are just Rust code and don't need any
9/// kind of CLI interface. It is useful, however, to provide a standardized
10/// interface for viewing and exporting models, and is used for Fornjot's
11/// example models and the testing infrastructure they are part of.
12///
13/// You might not want to use this struct directly. [`Instance::process_model`]
14/// provides a more high-level and convenient interface.
15///
16/// [`Instance::process_model`]: crate::Instance::process_model
17#[derive(clap::Parser)]
18pub struct Args {
19    /// Export model to this path
20    #[arg(short, long, value_name = "PATH")]
21    pub export: Option<PathBuf>,
22
23    /// How much the export can deviate from the original model
24    #[arg(short, long, value_parser = parse_tolerance)]
25    pub tolerance: Option<Tolerance>,
26
27    /// Ignore validation errors
28    #[arg(short, long)]
29    pub ignore_validation: bool,
30}
31
32impl Args {
33    /// Parse the command-line arguments
34    ///
35    /// Convenience method that saves the caller from having to import the
36    /// `clap::Parser` trait.
37    pub fn parse() -> Self {
38        <Self as clap::Parser>::parse()
39    }
40}
41
42fn parse_tolerance(input: &str) -> Result<Tolerance, ArgsError> {
43    let tolerance = f64::from_str(input)?;
44    let tolerance = Scalar::from_f64(tolerance);
45    let tolerance = Tolerance::from_scalar(tolerance)?;
46
47    Ok(tolerance)
48}
49
50#[derive(Debug, thiserror::Error)]
51pub enum ArgsError {
52    #[error("Error parsing tolerance")]
53    ParseTolerance(#[from] ParseFloatError),
54
55    #[error(transparent)]
56    InvalidTolerance(#[from] InvalidTolerance),
57}
