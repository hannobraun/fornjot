1/// RGBA color
2#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash, Ord, PartialOrd)]
3pub struct Color(pub [u8; 4]);

5/// Error for the color parsing issues
6#[derive(Debug, PartialEq, Eq)]
7pub struct ParseColorError;
8
9impl Default for Color {
10    fn default() -> Self {
11        // The default color is red. This is an arbitrary choice.
12        Self([255, 0, 0, 255])
13    }
14}

16impl From<[u8; 4]> for Color {
17    fn from(rgba: [u8; 4]) -> Self {

19    }
20}

22impl From<[u8; 3]> for Color {
23    fn from([r, g, b]: [u8; 3]) -> Self {

28impl From<[f64; 4]> for Color {
29    fn from(rgba: [f64; 4]) -> Self {
30        let rgba = rgba.map(|value| {
31            let value = value.clamp(0., 1.);
32            let value: u8 = (value * 255.0) as u8;
33            value
34        });
35
36        Self(rgba)
37    }
38}

40impl From<[f64; 3]> for Color {
41    fn from([r, g, b]: [f64; 3]) -> Self {

46impl TryFrom<&str> for Color {
47    type Error = ParseColorError;
48
49    fn try_from(hex_str: &str) -> Result<Self, Self::Error> {
50        let trimmed_hex = hex_str.trim_start_matches('#');
51        let len = trimmed_hex.len();
52
53        let parse_component = |start: usize, end: usize| -> Option<u8> {
54            u8::from_str_radix(&trimmed_hex[start..end], 16).ok()
55        };
56
57        match len {
58            6 => {
59                let r_opt = parse_component(0, 2);
60                let g_opt = parse_component(2, 4);
61                let b_opt = parse_component(4, 6);
62                if let (Some(r), Some(g), Some(b)) = (r_opt, g_opt, b_opt) {
63                    Ok(Self([r, g, b, 255]))
64                } else {
65                    Err(ParseColorError)
66                }
67            }
68            8 => {
69                let r_opt = parse_component(0, 2);
70                let g_opt = parse_component(2, 4);
71                let b_opt = parse_component(4, 6);
72                let a_opt = parse_component(6, 8);
73
74                // construction of color
75                if let (Some(r), Some(g), Some(b), Some(a)) =
76                    (r_opt, g_opt, b_opt, a_opt)
77                {
78                    Ok(Self([r, g, b, a]))
79                } else {
80                    Err(ParseColorError)
81                }
82            }
83            _ => Err(ParseColorError),
84        }
85    }
86}

88#[cfg(test)]
89mod tests {
90    use super::*;
91    use std::convert::TryFrom;
92
93    #[test]
94    fn test_hex_to_color() {
95        assert_eq!(Color::try_from("#FF0000"), Ok(Color([255, 0, 0, 255])));
96        assert_eq!(Color::try_from("FF0000"), Ok(Color([255, 0, 0, 255]))); // Test without #
97        assert_eq!(Color::try_from("#00FF0080"), Ok(Color([0, 255, 0, 128])));
98        assert_eq!(Color::try_from("00FF0080"), Ok(Color([0, 255, 0, 128]))); // Test without #
99        assert_eq!(
100            Color::try_from("#123456"),
101            Ok(Color([0x12, 0x34, 0x56, 255]))
102        );
103        assert_eq!(
104            Color::try_from("123456"),
105            Ok(Color([0x12, 0x34, 0x56, 255]))
106        );
107        assert_eq!(
108            Color::try_from("#123456AB"),
109            Ok(Color([0x12, 0x34, 0x56, 0xAB]))
110        );
111        assert_eq!(
112            Color::try_from("123456AB"),
113            Ok(Color([0x12, 0x34, 0x56, 0xAB]))
114        );
115        assert_eq!(Color::try_from("#ABC"), Err(ParseColorError)); // Invalid length
116        assert_eq!(Color::try_from("ABC"), Err(ParseColorError)); // Invalid length
117        assert_eq!(Color::try_from("#GGHHII"), Err(ParseColorError)); // Invalid hex characters
118        assert_eq!(Color::try_from("invalid"), Err(ParseColorError)); // Invalid input
119    }
